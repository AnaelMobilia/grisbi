00118 void display_tip ( gboolean force )
00119 {
00120     GtkWidget * checkbox;
00121     GtkWidget * dialog = NULL;
00122     GtkWidget *btn_back, *btn_forward, *btn_close;
00123     gchar *tmpstr;
00124 
00125     if ( !force && !etat.show_tip )
00126         return;
00127 
00128     etat.last_tip = CLAMP (etat.last_tip+1, 0, sizeof(tips)/sizeof(gpointer)-1);
00129 
00130     dialog = dialogue_special_no_run ( GTK_MESSAGE_INFO, GTK_BUTTONS_NONE,
00131                         make_hint ( _("Did you know that..."),
00132                         /* We use the grisbi-tips catalog */
00133                         g_dgettext(NULL, tips[etat.last_tip]) ) );
00134     gtk_window_set_modal ( GTK_WINDOW ( dialog ), FALSE );
00135 
00136     checkbox = gsb_automem_checkbutton_new ( _("Display tips at next start"), 
00137                         &(etat.show_tip), NULL, NULL );
00138     gtk_box_pack_start ( GTK_BOX ( GTK_DIALOG(dialog) -> vbox ), checkbox, FALSE, FALSE, 6 );
00139     gtk_widget_show ( checkbox );
00140 
00141     btn_back =    gtk_dialog_add_button (GTK_DIALOG(dialog), GTK_STOCK_GO_BACK, 1);
00142     btn_forward = gtk_dialog_add_button (GTK_DIALOG(dialog), GTK_STOCK_GO_FORWARD, 2);
00143     btn_close =   gtk_dialog_add_button (GTK_DIALOG(dialog), GTK_STOCK_CLOSE, 3);
00144  
00145     /* gtk_widget_set_size_request ( dialog, 450, -1 ); */
00146     /* We iterate as user can select several tips. */
00147     while ( TRUE )
00148     {
00149     if (etat.last_tip == sizeof(tips)/sizeof(gpointer)-1)
00150         gtk_widget_set_sensitive (btn_forward, FALSE);
00151     if (etat.last_tip == 0)
00152         gtk_widget_set_sensitive (btn_back, FALSE);
00153 
00154     switch ( gtk_dialog_run ( GTK_DIALOG(dialog) ) )
00155     {
00156         case 1:
00157         if ( etat.last_tip > 0 )
00158             etat.last_tip--;
00159         gtk_widget_set_sensitive (btn_forward, TRUE); 
00160         tmpstr = g_strconcat ( make_pango_attribut (
00161                         "size=\"larger\" weight=\"bold\"", _("Did you know that...") ),
00162                         "\n\n",
00163                         g_dgettext (NULL, tips[etat.last_tip] ),
00164                         NULL );
00165 
00166         gtk_label_set_markup ( GTK_LABEL ( GTK_MESSAGE_DIALOG(dialog) -> label ),
00167                         tmpstr );
00168         g_free ( tmpstr );
00169         break;
00170 
00171         case 2:
00172         if (etat.last_tip < sizeof(tips)/sizeof(gpointer)-1)
00173             etat.last_tip++;
00174         tmpstr = g_strconcat ( make_pango_attribut (
00175                         "size=\"larger\" weight=\"bold\"", _("Did you know that...") ),
00176                         "\n\n",
00177                         g_dgettext (NULL, tips[etat.last_tip] ),
00178                         NULL );
00179 
00180         gtk_label_set_markup ( GTK_LABEL ( GTK_MESSAGE_DIALOG(dialog) -> label ),
00181                         tmpstr );
00182         g_free ( tmpstr );
00183         gtk_widget_set_sensitive (btn_back, TRUE);
00184         break;
00185 
00186         default:
00187         gtk_widget_destroy ( dialog );
00188         return;
00189     }
00190     }
00191 }
