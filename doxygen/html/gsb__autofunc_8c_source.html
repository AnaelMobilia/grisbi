00081 /*
00082  * creates a new GtkEntry wich will modify the value according to the entry
00083  * but made for values in grisbi structure :
00084  * for each change, will call the corresponding given function : gsb_data_... ( number, string content )
00085  * ie the target function must be :
00086  *      (default_func) ( gint number_for_func,
00087  *                       gchar *string )
00088  * ex : gsb_data_account_set_name ( account_number, name )
00089  *
00090  * \param value a string to fill the entry
00091  * \param hook an optional function to execute as a handler if the
00092  *      entry's contents are modified.
00093  *      hook should be :
00094  *              gboolean hook ( GtkWidget *entry,
00095  *                              gpointer data )
00096  *
00097  * \param data An optional pointer to pass to hooks.
00098  * \param default_func The function to call to change the value in memory (function must be func ( number, string ) ) or NULL
00099  * \param number_for_func a gint wich we be used to call default_func (will be saved as g_object_set_data with "number_for_func")
00100  *                              that number can be changed with gsb_autofunc_entry_set_value
00101  *
00102  * \return a new GtkEntry
00103  * */
00104 GtkWidget *gsb_autofunc_entry_new ( const gchar *value,
00105                                     GCallback hook,
00106                                     gpointer data,
00107                                     GCallback default_func,
00108                                     gint number_for_func )
00109 {
00110     GtkWidget *entry;
00111 
00112     /* first, create and fill the entry */
00113     entry = gtk_entry_new ();
00114 
00115     if (value)
00116         gtk_entry_set_text ( GTK_ENTRY(entry), value );
00117 
00118     /* set the default func :
00119      * the func will be send to gsb_editable_set_text by the data,
00120      * the number_for_func will be set as data for object */
00121     g_object_set_data ( G_OBJECT (entry),
00122                         "number_for_func", GINT_TO_POINTER (number_for_func));
00123     if (default_func)
00124         g_object_set_data ( G_OBJECT ( entry ), "changed", 
00125                             (gpointer) g_signal_connect_after (G_OBJECT(entry), "changed",
00126                                                                G_CALLBACK (gsb_autofunc_entry_changed), default_func ));
00127     if ( hook )
00128         g_object_set_data ( G_OBJECT ( entry ), "changed-hook", 
00129                             (gpointer) g_signal_connect_after (G_OBJECT(entry), "changed",
00130                                                                G_CALLBACK (hook), data ));
00131     return entry;
00132 }

01300 /*
01301  * This create a combobox of currencies, this is basically the same as gsb_autofunc_combobox_new
01302  *      but work only with currencies
01303  *
01304  * for each change, will call the corresponding given function : gsb_data_... ( number, currency_number )
01305  * ie the target function must be :
01306  *      (default_func) ( gint number_for_func,
01307  *                       gint currency_number )
01308  * ex : gsb_data_account_set_currency ( account_number, currency_number )
01309  *
01310  * basically, that combobox is created with gsb_currency_make_combobox, so can use that functions to get the index
01311  *      if necessary
01312  *
01313  * \param set_name TRUE to show the name of the currencies in the combobox
01314  * \param currency_number the currency we want to show
01315  * \param hook an optional function to execute as a handler if the
01316  *      combobox changed.
01317  *      hook should be :
01318  *              gboolean hook ( GtkWidget *combobox,
01319  *                              gpointer data )
01320  *
01321  * \param data An optional pointer to pass to hooks.
01322  * \param default_func The function to call to change the value in memory (function must be func ( number, currency_number ) ) or NULL
01323  * \param number_for_func a gint wich we be used to call default_func (will be saved as g_object_set_data with "number_for_func")
01324  *                              that number can be changed with gsb_autofunc_currency_set_currency_number
01325  *
01326  * \return a new GtkComboBox 
01327  * */
01328 GtkWidget *gsb_autofunc_currency_new ( gboolean set_name,
01329                                        gint currency_number,
01330                                        GCallback hook,
01331                                        gpointer data,
01332                                        GCallback default_func,
01333                                        gint number_for_func )
01334 {
01335     GtkWidget *combobox;
01336 
01337     /* create and fill the combobox */
01338     combobox = gsb_currency_make_combobox (set_name);
01339 
01340     if ( combobox && currency_number )
01341         gsb_currency_set_combobox_history (combobox, currency_number);
01342 
01343     /* set the default func :
01344      * the func will be sent to gsb_autofunc_currency_changed by the data,
01345      * the number_for_func will be set as data for object */
01346     g_object_set_data ( G_OBJECT (combobox),
01347             "number_for_func", GINT_TO_POINTER (number_for_func));
01348 
01349     if (default_func)
01350     g_object_set_data ( G_OBJECT (combobox), "changed", 
01351                     (gpointer) g_signal_connect_after (G_OBJECT(combobox), "changed",
01352                     G_CALLBACK (gsb_autofunc_currency_changed), default_func ));
01353     if ( hook )
01354     g_object_set_data ( G_OBJECT (combobox), "changed-hook", 
01355                     (gpointer) g_signal_connect_after (G_OBJECT(combobox), "changed",
01356                     G_CALLBACK (hook), data ));
01357      return combobox;
01358 }
