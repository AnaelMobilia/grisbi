00293     GtkTreeModel *model;

00296     if (!reconcile_treeview)

00299     model = gtk_tree_view_get_model ( GTK_TREE_VIEW (reconcile_treeview));
00300 
00301     gtk_tree_store_clear (GTK_TREE_STORE(model));
00302 
00303     /* we make a tree_model containing the accounts,
00304      * and for each account, all the reconciles */
00305     tmp_list = gsb_data_account_get_list_accounts ();
00306     while (tmp_list)
00307     {
00308         gint account_number;
00309         GtkTreeIter account_iter;

00322                              -1 );
00323 
00324         /* for each account, get the concerned reconciles */
00325         reconcile_list = gsb_data_reconcile_get_reconcile_list ();
00326         while (reconcile_list)
00327         {
00328             gint reconcile_number;
00329 
00330             reconcile_number = gsb_data_reconcile_get_no_reconcile (reconcile_list -> data);
00331
