00515 static gboolean gsb_assistant_file_change_title ( GtkWidget *title_entry,
00516                                                   GtkWidget *filename_entry )
00517 {
00518     gchar *new_filename;
00519     gchar *last_filename;
00520     gchar *last_title;
00521 
00522     update_homepage_title (GTK_ENTRY (title_entry), NULL, 0, 0);
00523 
00524     /* first get the last content of the title to see if the filename
00525      * was automatically created, and in that case, we continue the automatic mode,
00526      * else we don't touch the filename entry */
00527     last_title = g_object_get_data ( G_OBJECT (title_entry),
00528                                      "last_title");
00529     last_filename = g_strconcat ( my_get_gsb_file_default_dir (),
00530                                   G_DIR_SEPARATOR_S,
00531                                   last_title,
00532                                   ".gsb",
00533                                   NULL );
00534     /* set the new -last title- */
00535     g_free (last_title);
00536     g_object_set_data ( G_OBJECT (title_entry),
00537                         "last_title", g_strdup (gtk_entry_get_text (GTK_ENTRY (title_entry))));
00538 
00539     if ( strcmp ( last_filename,
00540                   gtk_entry_get_text (GTK_ENTRY (filename_entry))))
00541     {
00542         /* there is a difference between the last title and the filename,
00543          * so juste free the memory and do nothing */
00544         g_free (last_filename);
00545         return FALSE;
00546     }
00547 
00548     /* ok, the filename is an automatic creation,
00549      * so we change it */
00550     g_free (last_filename);
00551 
00552     if ( strlen ( gtk_entry_get_text (GTK_ENTRY (title_entry) ) ) )
00553     {
00554         new_filename = g_strconcat ( my_get_gsb_file_default_dir (),
00555                                      G_DIR_SEPARATOR_S,
00556                                      gtk_entry_get_text (GTK_ENTRY (title_entry)),
00557                                      ".gsb",
00558                                      NULL );
00559     }
00560     else
00561     {
00562         new_filename = g_strconcat ( my_get_gsb_file_default_dir (),
00563                                      G_DIR_SEPARATOR_S,
00564                                      _("My accounts"),
00565                                      ".gsb",
00566                                      NULL );
00567     }
00568 
00569     gtk_entry_set_text ( GTK_ENTRY (filename_entry),
00570                          new_filename );
00571     g_free (new_filename);
00572 
00573     return FALSE;
