00685                                                 gsb_data_category_get_sub_category_number_by_name (

01388 void gsb_data_category_add_transaction_to_category ( gint transaction_number,
01389                                                      gint category_id,
01390                                                      gint sub_category_id )
01391 {
01392     struct_category *category;
01393     struct_sub_category *sub_category;
01394 
01395     /* if the transaction is a transfer or a split transaction, don't take it */
01396     if (gsb_data_transaction_get_split_of_transaction (transaction_number)
01397         ||
01398         gsb_data_transaction_get_contra_transaction_number (transaction_number) > 0)
01399         return;
01400 
01401     category = gsb_data_category_get_structure ( category_id );
01402     sub_category = gsb_data_category_get_sub_category_structure ( category_id,
01403                                                                   sub_category_id );
01404 
01405     /* should not happen, this is if the transaction has a categ wich doesn't exist
01406      * we show a debug warning and get without categ */
01407     if (!category)
01408     {
01409         gchar *tmpstr;
01410         tmpstr = g_strdup_printf ( _("The transaction %d has a category n∞%d and sub-category n∞%d but they don't exist."),
01411                                    transaction_number, 
01412                                    category_id,
01413                                    sub_category_id );
01414         warning_debug (tmpstr);
01415         g_free (tmpstr);
01416         category = empty_category;
01417     }
01418 
01419     /* ok, now category is on the structure or on empty_category */
01420     if ( category )
01421     {
01422         category -> category_nb_transactions ++;
01423     if ( ! gsb_data_transaction_get_split_of_transaction ( transaction_number ) )
01424         category -> category_balance = gsb_real_add ( category -> category_balance,
01425                         gsb_data_transaction_get_adjusted_amount_for_currency (
01426                         transaction_number, category_tree_currency (), -1));
01427     }
01428 
01429     /* if we were on empty category, no sub-category */
01430     if (category == empty_category)
01431         return;
01432 
01433     if ( sub_category )
01434     {
01435         sub_category -> sub_category_nb_transactions ++;
01436     if ( ! gsb_data_transaction_get_split_of_transaction ( transaction_number ) )
01437         sub_category -> sub_category_balance = gsb_real_add (
01438                         sub_category -> sub_category_balance,
01439                         gsb_data_transaction_get_adjusted_amount_for_currency (
01440                         transaction_number, category_tree_currency (), -1));
01441     }
01442     else
01443     {
01444         category -> category_nb_direct_transactions ++;
01445     if ( ! gsb_data_transaction_get_split_of_transaction ( transaction_number ) )
01446         category -> category_direct_balance = gsb_real_add (
01447                         category -> category_direct_balance,
01448                         gsb_data_transaction_get_adjusted_amount_for_currency (
01449                         transaction_number, category_tree_currency (), -1));
01450     }
