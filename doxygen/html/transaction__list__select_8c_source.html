00066

00070 
00071     /* if the selection didn't change, do nothing */
00072     if ( gsb_data_transaction_get_transaction_number (custom_list -> selected_row) == 
00073         transaction_number )

00077     if (custom_list -> selected_row)
00078         transaction_list_select_unselect ();

00083         record = custom_list -> visibles_rows [custom_list -> num_visibles_rows - custom_list -> nb_rows_by_transaction];

00085             record = NULL;

00087 
00088     if (!record)
00089     {
00090         GtkTreeIter iter;
00091 
00092         if (!transaction_model_get_transaction_iter ( &iter,
00093                                                       transaction_number,
00094                                                       0 ))
00095             return FALSE;
00096         record = iter.user_data;
00097     }
00098 
00099     /* si l'opération n'est pas visible on la rend visible */
00100     if ( ( record -> mother_row && record -> mother_row -> filtered_pos == -1 )
00101      ||
00102      record -> filtered_pos == -1 )
00103     {
00104         gint account_number;
00105 
00106         account_number = gsb_data_transaction_get_account_number ( transaction_number );
00107         gsb_data_account_set_r ( account_number, TRUE );
00108         gsb_menu_update_view_menu ( account_number );
00109         mise_a_jour_affichage_r ( TRUE );
00110     }
00111 
00112     return_value = transaction_list_select_record (record);
00113 
00114     /* if we are on a child, open the expander if necessary */
00115     path = gtk_tree_path_new ();
00116 
00117     if (record -> mother_row)
00118     {
00119         gtk_tree_path_append_index (path, record -> mother_row -> filtered_pos);
00120         gtk_tree_view_expand_row ( GTK_TREE_VIEW (gsb_transactions_list_get_tree_view ()),
00121                                    path, FALSE );
00122     }
00123     else
00124         gtk_tree_path_append_index (path, record -> filtered_pos);
00125 
00126     /* move the tree view to the selection */
00127     gtk_tree_view_scroll_to_cell ( GTK_TREE_VIEW (gsb_transactions_list_get_tree_view ()),
00128                                    path, NULL,
00129                                    FALSE, 0.0, 0.0 );
00130     gtk_tree_path_free (path);

00171 gboolean transaction_list_select_up ( gboolean into_children )
00172 {
00173     CustomRecord *record;
00174     GtkTreePath *path;
00175     CustomList *custom_list;
00176 
00177     custom_list = transaction_model_get_model ();
00178 
00179     g_return_val_if_fail ( custom_list != NULL, FALSE );
00180 
00181     record = custom_list -> selected_row;
00182     if (!record)
00183         /* there is no selection, should not happen
00184          * to avoid to block grisbi, set on the white line */
00185         return transaction_list_select (-1);
00186 
00187     /* check if first record */
00188     if (record -> filtered_pos == 0
00189         &&
00190         !record -> mother_row)
00191         return FALSE;
00192 
00193     /* go to previous record */
00194     if (record -> mother_row)
00195     {
00196         /* the current selection is a child, go up or on the mother */
00197         if (record -> pos == 0)
00198             /* we are on the first child, go to the mother */
00199             record = record -> mother_row -> transaction_records[0];
00200         else
00201             /* go to the prev child */
00202             record = record -> mother_row -> children_rows[record -> pos - 1];
00203     }
00204     else
00205     {
00206         /* the current selection is a mother, we go on the prev mother or the prev child */
00207 
00208         record = custom_list -> visibles_rows[record -> filtered_pos -1];
00209 
00210         /* if the upper row is an archive, do nothin */
00211         if (record -> what_is_line == IS_ARCHIVE)
00212             return FALSE;
00213 
00214         /* if the record is a mother, the expander is opened and into_children is TRUE,
00215          * go to the last child */
00216         path = gtk_tree_path_new ();
00217         gtk_tree_path_append_index (path, record -> filtered_pos);
00218 
00219         if (record -> number_of_children && into_children
00220             &&
00221             gtk_tree_view_row_expanded (GTK_TREE_VIEW (gsb_transactions_list_get_tree_view ()), path))
00222             record = record -> children_rows[record -> number_of_children - 1];
00223         else
00224             /* we want the mother, set record on the first row */
00225             record = record -> transaction_records[0];
00226 
00227         gtk_tree_path_free (path);
00228     }
00229 
00230     /* now can remove the selection and select the new one */
00231     transaction_list_select_unselect ();
00232     transaction_list_select_record (record);
00233 
00234     /* move the tree view to the selection */
00235     path = gtk_tree_path_new ();
00236     if (record -> mother_row)
00237         gtk_tree_path_append_index (path, record -> mother_row -> filtered_pos);
00238     gtk_tree_path_append_index (path, record -> filtered_pos);
00239     gtk_tree_view_scroll_to_cell ( GTK_TREE_VIEW (gsb_transactions_list_get_tree_view ()),
00240                                    path, NULL,
00241                                    FALSE, 0.0, 0.0 );
00242     gtk_tree_path_free (path);
00243     return TRUE;

00259 gboolean transaction_list_select_down ( gboolean into_children )
00260 {
00261     CustomRecord *record;
00262     GtkTreePath *end_path_list;
00263     GtkTreePath *path;
00264     CustomList *custom_list;
00265 
00266     custom_list = transaction_model_get_model ();
00267 
00268     g_return_val_if_fail ( custom_list != NULL, FALSE );
00269 
00270     /* the only way to forbid the move is if we are on the white line */
00271     record = custom_list -> selected_row;
00272     if (!record)
00273         /* there is no selection, should not happen
00274          * to avoid to block grisbi, set on the white line */
00275         return transaction_list_select (-1);
00276 
00277     /* the way is different between the selected transaction is a child or a mother */
00278     if (record -> mother_row)
00279     {
00280         /* we are on a child, go to the next child or to the next mother */
00281         if (record -> pos == (record -> mother_row -> number_of_children - 1))
00282             /* go to the next mother */
00283             record = custom_list -> visibles_rows[record -> mother_row -> filtered_pos + 1];
00284         else
00285             /* go to the next child */
00286             record = record -> mother_row -> children_rows[record -> pos + 1];
00287     }
00288     else
00289     {
00290         /* we are on a mother, go into the first child or the next mother */
00291 
00292         /* if we are on the white line, go away */
00293         if (gsb_data_transaction_get_transaction_number (record -> transaction_pointer) == -1)
00294             return FALSE;
00295 
00296         /* if we are on a split, and it is opened, and into_children i set, go
00297          * to the first child */
00298         if (record -> number_of_children && into_children)
00299         {
00300             path = gtk_tree_path_new ();
00301             /* set the path on the mother row wich have the expander */
00302             gtk_tree_path_append_index (path, record -> children_rows[0] -> mother_row -> filtered_pos);
00303         }
00304         else
00305             path = NULL;
00306 
00307         if (path
00308             &&
00309             gtk_tree_view_row_expanded (GTK_TREE_VIEW (gsb_transactions_list_get_tree_view ()), path))
00310         {
00311             /* go to the first child */
00312             record = record -> children_rows[0];
00313             gtk_tree_path_free (path);
00314         }
00315         else
00316             /* go to the next transaction mother */
00317             record = custom_list -> visibles_rows[record -> filtered_pos + custom_list -> nb_rows_by_transaction];
00318     }
00319 
00320     /* now can remove the selection and select the new one */
00321     transaction_list_select_unselect ();
00322     transaction_list_select_record (record);
00323 
00324     /* move the tree view to the selection, we need to check the visible range and
00325      * all that stuff because gtk_tree_view_scroll_to_cell set the rows at the upper of the tree view
00326      * instead of keeping them at the bottom, when transaction goes down beside the bottom of the tree view */
00327     if ( gtk_tree_view_get_visible_range ( GTK_TREE_VIEW (gsb_transactions_list_get_tree_view ()),
00328                                            NULL,
00329                                            &end_path_list ))
00330     {
00331         path = gtk_tree_path_new ();
00332 
00333         if (record -> mother_row)
00334         {
00335             gtk_tree_path_append_index (path, record -> mother_row -> filtered_pos);
00336             gtk_tree_path_append_index (path, record -> filtered_pos);
00337         }
00338         else
00339             /* check with the last visible line of the transaction */
00340             gtk_tree_path_append_index (path, record -> transaction_records[custom_list -> nb_rows_by_transaction - 1] -> filtered_pos);
00341 
00342         if (gtk_tree_path_compare (end_path_list, path) == -1)
00343             /* the last line of the transaction is above the last visible row */
00344             gtk_tree_view_scroll_to_cell ( GTK_TREE_VIEW (gsb_transactions_list_get_tree_view ()),
00345                                            path, NULL,
00346                                            TRUE, 1.0, 0.0 );
00347 
00348         gtk_tree_path_free (path);
00349         gtk_tree_path_free (end_path_list);
00350     }
00351     return TRUE;
00352 }
