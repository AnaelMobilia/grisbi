00579 gint gsb_data_archive_get_from_date ( const GDate *date )
00580 {
00581     GSList *tmp_list;
00582     gint return_value = 0;
00583 
00584     if (!date)
00585         return 0;
00586 
00587     tmp_list = archive_list;
00588     while (tmp_list)
00589     {
00590         struct_archive *archive;
00591 
00592         archive = tmp_list -> data;
00593 
00594         /* check the archive only if the dates are valid */
00595         if (archive -> beginning_date && archive -> end_date)
00596         {
00597             if ( g_date_compare ( date, archive -> beginning_date) >= 0
00598                  &&
00599                  g_date_compare ( date, archive -> end_date) <= 0 )
00600             {
00601                 if (return_value)
00602                     return_value = -1;
00603                 else
00604                     return_value = archive -> archive_number;
00605             }
00606         }
00607         tmp_list = tmp_list -> next;
00608     }
00609     return return_value;
00610 }
