00084 gboolean gsb_scheduler_increase_scheduled ( gint scheduled_number )
00085 {
00086     GDate *new_date;
00087 
00088     gchar* tmpstr = g_strdup_printf ("%d", scheduled_number);
00089     devel_debug ( tmpstr );
00090     g_free ( tmpstr );
00091 
00092     g_return_val_if_fail ( g_date_valid (gsb_data_scheduled_get_date (scheduled_number)), TRUE );
00093 
00094     /* increase the date of the scheduled_transaction */
00095     new_date = gsb_scheduler_get_next_date ( scheduled_number,
00096                                              gsb_data_scheduled_get_date (scheduled_number));
00097 
00098     /* we continue to work only if new_date is not null (null mean reach the end) */
00099     if (new_date)
00100     {
00101         /* set the new date */
00102         gsb_data_scheduled_set_date ( scheduled_number, new_date);
00103 
00104         if ( gsb_data_scheduled_get_split_of_scheduled ( scheduled_number ))
00105         {
00106             GSList *children_numbers_list;
00107 
00108             /* if there is some children, set the new date too */
00109             children_numbers_list = gsb_data_scheduled_get_children ( scheduled_number, TRUE );
00110 
00111             while ( children_numbers_list )
00112             {
00113                 gint child_number;
00114 
00115                 child_number = GPOINTER_TO_INT ( children_numbers_list -> data );
00116 
00117                 gsb_data_scheduled_set_date ( child_number,
00118                                               new_date );
00119 
00120                 children_numbers_list = children_numbers_list -> next;
00121             }
00122             g_slist_free (children_numbers_list);
00123         }
00124         g_date_free (new_date);
00125     }
00126     else
00127     {
00128         /* the scheduled transaction is over, we remove it */
00129         /* update the main page */
00130         gsb_main_page_update_finished_scheduled_transactions (scheduled_number);
00131 
00132         /* remove the scheduled transaction */
00133         /* !! important to remove first from the list... */
00134         gsb_scheduler_list_remove_transaction_from_list ( scheduled_number );

00136         return FALSE;

00152 GDate *gsb_scheduler_get_next_date ( gint scheduled_number,
00153                                      const GDate *date )
00154 {
00155     GDate *return_date;
00156 
00157     if ( !scheduled_number
00158          ||
00159          !gsb_data_scheduled_get_frequency (scheduled_number)
00160          ||
00161          !date
00162          ||
00163          !g_date_valid (date))
00164         return NULL;
00165 
00166     /* we don't change the initial date */
00167     return_date = gsb_date_copy (date);
00168 
00169     switch (gsb_data_scheduled_get_frequency (scheduled_number))
00170     {
00171         case SCHEDULER_PERIODICITY_ONCE_VIEW:
00172             return NULL;
00173             break;
00174 
00175         case SCHEDULER_PERIODICITY_WEEK_VIEW:
00176             g_date_add_days ( return_date, 7 );
00177             /* FIXME : there were a bug in gtk and we had to add 0 month to have the good date,
00178              * it seems fixed but we should wait the stable debian is upgraded to
00179              * remove that [26/10/2008] */
00180             g_date_add_months ( return_date, 0 );
00181             break;
00182 
00183         case SCHEDULER_PERIODICITY_MONTH_VIEW:
00184             g_date_add_months ( return_date, 1 );
00185             break;
00186 
00187         case SCHEDULER_PERIODICITY_TWO_MONTHS_VIEW:
00188             g_date_add_months ( return_date, 2 );
00189             break;
00190 
00191         case SCHEDULER_PERIODICITY_TRIMESTER_VIEW:
00192             g_date_add_months ( return_date, 3 );
00193             break;
00194 
00195         case SCHEDULER_PERIODICITY_YEAR_VIEW:
00196             g_date_add_years ( return_date, 1 );
00197             break;
00198 
00199         case SCHEDULER_PERIODICITY_CUSTOM_VIEW:
00200             if ( gsb_data_scheduled_get_user_entry (scheduled_number) <= 0 )
00201             {
00202                 g_date_free (return_date);
00203                 return NULL;
00204             }
00205 
00206             switch (gsb_data_scheduled_get_user_interval (scheduled_number))
00207             {
00208                 case PERIODICITY_DAYS:
00209                     g_date_add_days ( return_date, 
00210                                       gsb_data_scheduled_get_user_entry (scheduled_number));
00211                     /* FIXME : there were a bug in gtk and we had to add 0 month to have the good date,
00212                      * it seems fixed but we should wait the stable debian is upgraded to
00213                      * remove that [26/10/2008] */
00214                     g_date_add_months ( return_date, 0 );
00215                     break;
00216 
00217                 case PERIODICITY_WEEKS:
00218                     g_date_add_days ( return_date, 
00219                                       gsb_data_scheduled_get_user_entry (scheduled_number) * 7 );
00220                     g_date_add_months ( return_date, 0 );
00221                     break;
00222 
00223                 case PERIODICITY_MONTHS:
00224                     g_date_add_months ( return_date,
00225                                         gsb_data_scheduled_get_user_entry (scheduled_number));
00226                     break;
00227 
00228                 case PERIODICITY_YEARS:
00229                     g_date_add_years ( return_date,
00230                                        gsb_data_scheduled_get_user_entry (scheduled_number));
00231                     g_date_add_months ( return_date, 0 );
00232                     break;
00233             }
00234             break;
00235     }
00236 
00237     if ( gsb_data_scheduled_get_limit_date (scheduled_number)
00238          &&
00239          g_date_compare ( return_date,
00240                           gsb_data_scheduled_get_limit_date (scheduled_number)) > 0 )
00241     {
00242         g_date_free (return_date);
00243         return_date = NULL;
00244     }
00245     
00246     return ( return_date );
00247 }

00368     /* if category is set, it's a normal category */

00378         return TRUE;
00379     }
00380 
00381     if ( gsb_data_scheduled_get_split_of_scheduled (scheduled_number))

00386 
00387         gsb_data_transaction_set_split_of_transaction ( transaction_number,
00388                                                             1 );
00389     }
00390     else
00391     {
00392         /* it's not a split of transaction and not a normal category so it's a transfer
00393          * except if the target account is -1 then it's a
00394          * transaction with no category */
00395 
00396         if ( gsb_data_scheduled_get_account_number_transfer (scheduled_number) != -1 )
00397         {
00398             gint contra_transaction_number;
00399

00404                                                                 gsb_data_scheduled_get_contra_method_of_payment_number (scheduled_number));
00405         }
00406     }

00456 void gsb_scheduler_check_scheduled_transactions_time_limit ( void )
00457 {
00458     GDate *date;
00459     GSList *tmp_list;
00460     gboolean automatic_transactions_taken = FALSE;
00461 
00462     devel_debug (NULL);
00463 
00464     /* the scheduled transactions to take will be check here,
00465      * but the scheduled transactions taken will be add to the already appended ones */
00466 
00467     scheduled_transactions_to_take = NULL;
00468 
00469     /* get the date today + nb_days_before_scheduled */
00470 
00471     /* the date untill we execute the scheduled transactions is :
00472      * - either today + nb_days_before_scheduled if warn n days before the scheduled
00473      * - either the end of the month in nb_days_before_scheduled days (so current month or next month)
00474      *   */
00475     date = gdate_today ();
00476     g_date_add_days ( date,
00477                       nb_days_before_scheduled );
00478     /* now date is in nb_days_before_scheduled, if we want the transactions of the month,
00479      * we change date to the end of its month */
00480     if (execute_scheduled_of_month)
00481     {
00482         gint last_day;
00483         
00484         last_day = g_date_get_days_in_month ( g_date_get_month (date),
00485                                               g_date_get_year (date));
00486         g_date_set_day (date, last_day);
00487     }
00488 
00489     /* check all the scheduled transactions,
00490      * if automatic, it's taken
00491      * if manual, appended into scheduled_transactions_to_take */
00492     tmp_list = gsb_data_scheduled_get_scheduled_list ();
00493 
00494     while ( tmp_list )
00495     {
00496         gint scheduled_number;
00497 
00498         scheduled_number = gsb_data_scheduled_get_scheduled_number (tmp_list -> data);
00499 
00500         /* we check that scheduled transaction only if it's not a child of a split */
00501         if ( !gsb_data_scheduled_get_mother_scheduled_number (scheduled_number)
00502              &&
00503              gsb_data_scheduled_get_date (scheduled_number)
00504              &&
00505              g_date_compare ( gsb_data_scheduled_get_date (scheduled_number),
00506                               date ) <= 0 )
00507         {
00508             if ( gsb_data_scheduled_get_automatic_scheduled (scheduled_number))
00509             {
00510                 /* this is an automatic scheduled, we get it */
00511                 gint transaction_number;
00512 
00513                 /* take automatically the scheduled transaction untill today */
00514                 transaction_number = gsb_scheduler_create_transaction_from_scheduled_transaction (scheduled_number,
00515                                                                                                   0 );
00516                 if ( gsb_data_scheduled_get_split_of_scheduled (scheduled_number))
00517                     gsb_scheduler_execute_children_of_scheduled_transaction ( scheduled_number,
00518                                                                               transaction_number );
00519 
00520                 scheduled_transactions_taken = g_slist_append ( scheduled_transactions_taken,
00521                                                                 GINT_TO_POINTER (transaction_number));
00522                 automatic_transactions_taken = TRUE;
00523 
00524                 /* set the scheduled transaction to the next date,
00525                  * if it's not finished, we check them again if it need to be
00526                  * executed more than one time (the easiest way is to check
00527                  * all again, i don't think it will have thousand of scheduled transactions, 
00528                  * so no much waste of time...) */
00529                 if (gsb_scheduler_increase_scheduled (scheduled_number))
00530                 {
00531                     scheduled_transactions_to_take = NULL;
00532                     tmp_list = gsb_data_scheduled_get_scheduled_list ();
00533                 }
00534                 else
00535                     /* the scheduled is finish, so we needn't to check it again ... */
00536                     tmp_list = tmp_list -> next;
00537             }
00538             else
00539             {
00540                 /* it's a manual scheduled transaction, we put it in the slist */
00541                 scheduled_transactions_to_take = g_slist_append ( scheduled_transactions_to_take ,
00542                                                                   GINT_TO_POINTER (scheduled_number));
00543                 tmp_list = tmp_list -> next;
00544             }
00545         }
00546         else
00547             tmp_list = tmp_list -> next;
00548     }
00549 
00550     if ( automatic_transactions_taken )
00551     {
00552         mise_a_jour_liste_echeances_auto_accueil = 1;
00553         if ( etat.modification_fichier == 0 )
00554         modification_fichier ( TRUE );
00555     }
00556 
00557     if ( scheduled_transactions_to_take )
00558         mise_a_jour_liste_echeances_manuelles_accueil = 1;
00559 
00560     g_date_free ( date );
00561 }
