00282 struct_scheduled *gsb_data_scheduled_get_scheduled_by_no ( gint scheduled_number )
00283 {
00284     GSList *scheduled_list_tmp;
00285 
00286     /* check first if the scheduled is in the buffer */
00287 
00288     if ( scheduled_buffer[0]
00289          &&
00290          scheduled_buffer[0] -> scheduled_number == scheduled_number )
00291         return scheduled_buffer[0];
00292 
00293     if ( scheduled_buffer[1]
00294          &&
00295          scheduled_buffer[1] -> scheduled_number == scheduled_number )
00296         return scheduled_buffer[1];
00297 
00298     if ( scheduled_number < 0 )
00299         scheduled_list_tmp = white_scheduled_list;
00300     else
00301         scheduled_list_tmp = scheduled_list;
00302 
00303     while ( scheduled_list_tmp )
00304     {
00305         struct_scheduled *scheduled;
00306 
00307         scheduled = scheduled_list_tmp -> data;
00308 
00309         if ( scheduled -> scheduled_number == scheduled_number )
00310         {
00311             gsb_data_scheduled_save_scheduled_pointer ( scheduled );
00312             return scheduled;
00313         }
00314 
00315         scheduled_list_tmp = scheduled_list_tmp -> next;
00316     }
00317 
00318     /* here, we didn't find any scheduled with that number */
00319 
00320     return NULL;
00321 }

01124 gboolean gsb_data_scheduled_is_transfer ( gint scheduled_number )
01125 {
01126     struct_scheduled *scheduled;
01127 
01128     scheduled = gsb_data_scheduled_get_scheduled_by_no ( scheduled_number);
01129 
01130     if ( !scheduled )
01131         return FALSE;
01132 
01133     /* it's more complex than for a transaction, because no transaction number to transfer,
01134      * we have only account_number_transfer, and it can be 0...
01135      * first, need to check if it's -1 => neither transfer, neither categ */
01136 
01137     if (scheduled -> account_number_transfer == -1)
01138         return FALSE;
01139 
01140     /* next, if account_number_transfer is > 0, it's a transfer */
01141     if (scheduled -> account_number_transfer > 0)
01142         return TRUE;
01143     
01144     /* ok, now we have an account_number_transfer at 0, so it can be
01145      * a normal scheduled transactions (with categs), or split */
01146     if (scheduled -> category_number
01147         ||
01148         scheduled -> split_of_scheduled)
01149         return FALSE;
01150     return TRUE;
01151 }

01601 {

01603 
01604     scheduled = gsb_data_scheduled_get_scheduled_by_no ( scheduled_number);
01605
