00445     /* load the file */
00446

00448                                &file_content,

00450                                NULL ))
00451     {
00452         GMarkupParser *markup_parser = g_malloc0 (sizeof (GMarkupParser));
00453         GMarkupParseContext *context;
00454 
00455         /* check if it's a good file */
00456         if ( !gsb_file_others_check_file ( file_content,
00457                                            origin ))

00459             g_free (file_content);
