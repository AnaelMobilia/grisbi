00366     gint i, max = 0;
00367 
00368     for ( i = 0 ; separators[i] ; i++ )

00372

00608

00611     list = csv_get_next_line ( &contents, separator );

00621         gchar * value = list -> data;
00622 
00623         for ( field = 0 ; csv_fields [ field ] . name != NULL ; field ++ )
00624         {
00625             if ( strlen ( value ) > 1 &&
00626                  strlen ( csv_fields [ field ] . name ) > 1 )

00628             if ( strlen ( csv_fields [ field ] . alias ) > 1

00635                 if ( !default_config [ i ] )

00637                     default_config [ i ] = field;

00643              safe_contains ( _( csv_fields [ field ] . name ), value ) )
00644             {
00645                 if ( !default_config [ i ] )
00646                 {
00647                     default_config [ i ] = field;
00648                     etat.csv_skipped_lines [ 0 ] = 1;
00649                }
00650             }
00651         }
00652         }
00653 
00654         list = list -> next;
00655     }

00657     csv_skip_lines ( &contents, 3, separator );

00678                      ! default_config [ i ] )

00698                 }
00699                     }
00700                     else
00701                     {
00702                 if ( ! default_config [ i ] )
00703                 {
00704                     default_config [ i ] = 12; /* Negative debit */
00705                 }
00706                 else if ( default_config [ i ] == 14 )
00707                 {
00708                     default_config [ i ] = 11; /* Neutral amount */
00709                 }
