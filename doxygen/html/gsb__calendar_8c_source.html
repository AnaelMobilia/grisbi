00098 gboolean gsb_calendar_update ( void )
00099 {
00100     time_t temps;
00101     GSList *tmp_list;
00102     gint calendar_month;
00103         gint calendar_year;
00104 
00105     gtk_calendar_clear_marks ( GTK_CALENDAR ( scheduled_calendar ));
00106 
00107     /* select the current day */
00108     time ( &temps );
00109 
00110     if ( ( localtime ( &temps ) -> tm_mon == GTK_CALENDAR ( scheduled_calendar ) -> month )
00111          &&
00112          ( ( localtime ( &temps ) -> tm_year + 1900 ) == GTK_CALENDAR ( scheduled_calendar ) -> year ) )
00113         gtk_calendar_select_day ( GTK_CALENDAR ( scheduled_calendar ),
00114                                   localtime ( &temps ) -> tm_mday );
00115     else
00116         gtk_calendar_select_day ( GTK_CALENDAR ( scheduled_calendar ),
00117                                   FALSE );
00118 
00119     calendar_month = GTK_CALENDAR ( scheduled_calendar ) -> month + 1;
00120         calendar_year = GTK_CALENDAR ( scheduled_calendar ) -> year + 25;
00121 
00122     /* check the scheduled transactions and bold them in the calendar */
00123     tmp_list = gsb_data_scheduled_get_scheduled_list ();
00124 
00125     while ( tmp_list )
00126     {
00127         GDate *tmp_date;
00128         gint scheduled_number;
00129 
00130         scheduled_number = gsb_data_scheduled_get_scheduled_number (tmp_list -> data);
00131 
00132         tmp_date = gsb_date_copy (gsb_data_scheduled_get_date (scheduled_number));
00133 
00134         while (tmp_date && g_date_get_month (tmp_date) == calendar_month && g_date_get_year (tmp_date) < calendar_year)
00135         {
00136             GDate *new_date;
00137 
00138             gtk_calendar_mark_day ( GTK_CALENDAR ( scheduled_calendar ),
00139                                     g_date_get_day (tmp_date));
00140             new_date = gsb_scheduler_get_next_date (scheduled_number, tmp_date);
00141             g_free (tmp_date);
00142             tmp_date = new_date;
00143         }
00144         tmp_list = tmp_list -> next;
00145     }
00146     return FALSE;
00147 }
