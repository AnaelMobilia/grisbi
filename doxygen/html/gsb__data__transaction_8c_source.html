00745 gsb_real gsb_data_transaction_get_adjusted_amount_for_currency ( gint transaction_number,
00746                         gint return_currency_number,
00747                         gint return_exponent )
00748 {
00749     struct_transaction *transaction;
00750     gsb_real amount = null_real;
00751     gint link_number;
00752 
00753     if (return_exponent == -1)
00754         return_exponent = gsb_data_currency_get_floating_point (return_currency_number);
00755 
00756     transaction = gsb_data_transaction_get_transaction_by_no ( transaction_number);
00757 
00758     if ( ! (transaction && return_currency_number ) )
00759         return gsb_real_adjust_exponent  ( null_real, return_exponent );
00760 
00761     /* if the transaction currency is the same of the account's one,
00762      * we just return the transaction's amount */
00763     if ( transaction -> currency_number == return_currency_number )
00764         return gsb_real_adjust_exponent  ( transaction -> transaction_amount,
00765                                            return_exponent );
00766 /* printf ("get_adjusted_amount transaction_number = %d transaction -> currency_number = %d return_currency_number = %d\n",
00767     transaction_number, transaction -> currency_number, return_currency_number); */
00768     /* now we can adjust the amount */
00769         /* the exchange is saved in the transaction itself */
00770     if ( transaction -> exchange_rate.mantissa )
00771     {
00772         if ( transaction -> change_between_account_and_transaction )
00773             amount = gsb_real_div ( transaction -> transaction_amount,
00774                         transaction -> exchange_rate );
00775         else
00776             amount = gsb_real_mul ( transaction -> transaction_amount,
00777                         transaction -> exchange_rate );
00778 
00779         /* The costs are still deducted from the transaction. In case of internal transfer there is no charge. */
00780         amount = gsb_real_sub (amount, transaction -> exchange_fees);
00781     }
00782     else if ( (link_number = gsb_data_currency_link_search ( transaction -> currency_number,
00783                                                         return_currency_number ) ) )
00784     {
00785         /* there is a hard link between the transaction currency and the return currency */
00786         if ( gsb_data_currency_link_get_first_currency (link_number) == transaction -> currency_number)
00787             amount = gsb_real_mul ( transaction -> transaction_amount,
00788                         gsb_data_currency_link_get_change_rate (link_number));
00789         else
00790             amount = gsb_real_div ( transaction -> transaction_amount,
00791                         gsb_data_currency_link_get_change_rate (link_number));
00792 
00793         /* The costs are still deducted from the transaction. In case of internal transfer there is no charge. */
00794         amount = gsb_real_sub (amount, transaction -> exchange_fees);
00795     }

00797     {
00798         gsb_real current_exchange;

00802                         transaction -> currency_number,

00815                         current_exchange );
