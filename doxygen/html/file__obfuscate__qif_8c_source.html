00138 static void file_obfuscate_qif_start ( gchar *filename )
00139 {
00140     gchar *file_content;
00141     gchar *new_filename;
00142     gint position = 0;
00143     gint position_in_line = 0;
00144     gchar *tmpstr;
00145 
00146     if (!g_file_get_contents (filename,
00147                               &file_content, NULL, NULL ))
00148     {
00149         dialogue_error (_("Cannot open the QIF file."));
00150         return;
00151     }
00152 
00153     if (file_content[0] != '!')
00154     {
00155         dialogue_error (_("This file doesn't seem to be a QIF file,\noperation canceled."));
00156         g_free (file_content);
00157         return;
00158     }
00159 
00160     /* the qif file is opened, we can change the content */
00161     /* it's not too complex :
00162      * we need to keep the first letter, if it's ^, go to the next line
00163      * if the second letter is [, we need to keep the content of the line, it's the name of an account
00164      *          and we need it for transfer
00165      * same if the first letter is C (it's the mark for transaction) or ! or D (don't touch the date...)
00166      * after that, we change letter by other letter, and digits by other digits, but keep all the rest (. , ' ...)
00167      * */
00168 
00169     do
00170     {
00171         /* first check when we have to pass the line */
00172         switch (position_in_line)
00173         {
00174             case 0:
00175                 switch (file_content[position])
00176                 {
00177                     case '!':
00178                     case '^':
00179                     case 'C':
00180                     case 'D':
00181                         file_obfuscate_qif_next_line (file_content,
00182                                                       &position,
00183                                                       &position_in_line);
00184                         continue;
00185                         break;
00186                     default:
00187                         /* we are on the first col but don't pass this line so go to the second col
00188                          * and don't break here to check just after */
00189                         position++;
00190                         position_in_line++;
00191                 }
00192 
00193             case 1:
00194                 /* we are on the 2nd col, we pass this line if it's a transfer */
00195                 if (file_content[position] == '[')
00196                 {
00197                     file_obfuscate_qif_next_line (file_content,
00198                                                   &position,
00199                                                   &position_in_line);
00200                     continue;
00201                 }
00202                 break;
00203         }
00204 
00205         /* now we are on the 2nd col or more
00206          * if we are at the end of line, go to next line
00207          * in windows format, we can have \r before the \n */
00208         if (file_content[position] == '\r')
00209         {
00210             position++;
00211             position_in_line++;
00212         }
00213         if (file_content[position] == '\n')
00214         {
00215             position++;
00216             position_in_line = 0;
00217             continue;
00218         }
00219 
00220         /* ok, so now we need to change the content of the position, only for
00221          * digits or characters */
00222         if (g_ascii_isdigit (file_content[position]))
00223             file_content[position] = g_random_int_range ('0', '9');
00224         else
00225             if (g_ascii_isalpha (file_content[position]))
00226                 file_content[position] = g_random_int_range ('a', 'z');
00227 
00228         position++;
00229         position_in_line++;
00230     }
00231     while (file_content[position] != 0);
00232 
00233     filename[strlen(filename) - 4] = 0;
00234     new_filename = g_strconcat (filename, "-anonymized.qif", NULL);
00235 
00236     if (g_file_set_contents ( new_filename,
00237                               file_content,
00238                               -1, NULL ))
00239     {
00240         tmpstr = g_strdup_printf (_("File '%s' successfully saved"),
00241                                   new_filename );
00242         dialogue (tmpstr);
00243         g_free (tmpstr);
00244     }
00245     else
00246     {
00247         tmpstr = g_strdup_printf (_("File '%s' could not be saved"),
00248                                   new_filename );
00249         dialogue_error (tmpstr);
00250         g_free (tmpstr);
00251     }
00252 
00253     g_free (new_filename);
00254     g_free (file_content);
00255 }
