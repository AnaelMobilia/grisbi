00501 gboolean gsb_data_currency_link_check_for_invalid ( gint currency_link_number )
00502 {
00503     struct_currency_link *currency_link;
00504     GSList *tmp_list;
00505 
00506     currency_link = gsb_data_currency_link_get_structure ( currency_link_number );
00507 
00508     if (!currency_link)
00509         return FALSE;
00510 
00511     /* first check : if the 2 currencies are identical */
00512 
00513     if ( currency_link -> first_currency == currency_link -> second_currency)
00514     {
00515         currency_link -> invalid_link = TRUE;
00516         return TRUE;
00517     }
00518 
00519     /* second check : if that link exists already */
00520 
00521     tmp_list = currency_link_list;
00522 
00523     while (tmp_list)
00524     {
00525         struct_currency_link *tmp_currency_link;
00526 
00527         tmp_currency_link = tmp_list -> data;
00528 
00529         if ( tmp_currency_link -> currency_link_number != currency_link -> currency_link_number
00530              &&
00531              (( tmp_currency_link -> first_currency == currency_link -> first_currency
00532                 &&
00533                 tmp_currency_link -> second_currency == currency_link -> second_currency )
00534               ||
00535               ( tmp_currency_link -> first_currency == currency_link -> second_currency
00536                 &&
00537                 tmp_currency_link -> second_currency == currency_link -> first_currency )))
00538         {
00539             currency_link -> invalid_link = TRUE;
00540             return TRUE;
00541         }
00542 
00543         tmp_list = tmp_list -> next;
00544     }
00545 
00546     /* it's ok, the link is valid */
00547     currency_link -> invalid_link = FALSE;
00548 
00549     return FALSE;
00550 }

00561 gint gsb_data_currency_link_search ( gint currency_1,
00562                         gint currency_2 )
00563 {
00564     GSList *tmp_list;
00565 
00566     if (!currency_1
00567         ||
00568         !currency_2 )
00569         return 0;
00570 
00571     if ( currency_1 == currency_2 )
00572         return -1;
00573 
00574     tmp_list = currency_link_list;
00575 
00576     while (tmp_list)
00577     {
00578         struct_currency_link *tmp_currency_link;
00579 
00580         tmp_currency_link = tmp_list -> data;
00581 
00582         if ( !tmp_currency_link -> invalid_link
00583              &&
00584              (( tmp_currency_link -> first_currency == currency_1
00585                 &&
00586                 tmp_currency_link -> second_currency == currency_2 )
00587               ||
00588               ( tmp_currency_link -> first_currency == currency_2
00589                 &&
00590                 tmp_currency_link -> second_currency == currency_1 )))
00591             return tmp_currency_link -> currency_link_number;
00592 
00593         tmp_list = tmp_list -> next;
00594     }
00595     return 0;
