00142 void set_popup_position (GtkMenu *menu, gint *x, gint *y, gboolean *push_in, gpointer user_data)
00143 {
00144     GtkWidget *widget;
00145     GtkRequisition requisition;
00146     gint screen_width, menu_xpos, menu_ypos, menu_width;
00147 
00148     widget = GTK_WIDGET (user_data);
00149 
00150     gtk_widget_get_child_requisition (GTK_WIDGET (menu), &requisition);
00151     menu_width = requisition.width;
00152 
00153     gdk_window_get_origin (widget->window, &menu_xpos, &menu_ypos);
00154 
00155     menu_xpos += widget->allocation.x;
00156     menu_ypos += widget->allocation.y + widget->allocation.height - 2;
00157 
00158     if (gtk_widget_get_direction (widget) == GTK_TEXT_DIR_RTL)
00159         menu_xpos = menu_xpos + widget->allocation.width - menu_width;
00160 
00161     /* Clamp the position on screen */
00162     screen_width = gdk_screen_get_width (gtk_widget_get_screen (widget));
00163 
00164     if (menu_xpos < 0)
00165         menu_xpos = 0;
00166     else if ((menu_xpos + menu_width) > screen_width)
00167         menu_xpos -= ((menu_xpos + menu_width) - screen_width);
00168 
00169     *x = menu_xpos;
00170     *y = menu_ypos;
00171     *push_in = TRUE;
00172 }
