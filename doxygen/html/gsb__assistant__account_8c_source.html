00407 static gboolean gsb_assistant_account_enter_page_finish ( GtkWidget * assistant, gint new_page )
00408 {
00409     GtkWidget * account_entry_name = g_object_get_data ( G_OBJECT (assistant), "account_entry_name" );
00410     gchar * default_name = NULL;
00411     gint account_type = GPOINTER_TO_INT ( g_object_get_data ( G_OBJECT (assistant), "account_kind" ) );
00412 
00413     switch ( account_type )
00414     {
00415         case GSB_TYPE_BANK:
00416             if ( gsb_bank_list_get_bank_number ( account_combobox_bank ) >= 0 )
00417             {
00418                 const gchar * bank_name;
00419 
00420                 bank_name = gsb_data_bank_get_name ( gsb_bank_list_get_bank_number ( account_combobox_bank ) );
00421                 /* Do not use a bank name if no bank is set or if its
00422                  * name has not been changed. */
00423                 if ( bank_name && strcmp ( bank_name, _("New bank") ) )
00424                     default_name = g_strdup_printf ( _("%s account"), bank_name );
00425                 else
00426                     default_name = g_strdup ( _("Bank account" ) );
00427             }
00428             break;
00429 
00430         case GSB_TYPE_CASH:
00431             default_name = g_strdup_printf ( _("Cashier") );
00432             break;
00433 
00434         case GSB_TYPE_ASSET:
00435             default_name = g_strdup_printf ( _("Assets account") );
00436             break;
00437 
00438         case GSB_TYPE_LIABILITIES:
00439             default_name = g_strdup_printf ( _("Liabilities account"));
00440             break;
00441     }
00442 
00443     gtk_entry_set_text ( GTK_ENTRY ( account_entry_name ), default_name );
00444     if ( default_name )
00445         g_free ( default_name );
00446 
00447     return FALSE;
00448 }
