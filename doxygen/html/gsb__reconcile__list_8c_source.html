00140 
00141     /* ***** second part : specific tests on method of payment */
00142     /* if we are here, all the previous test passed, so we know it's 2 different transactions

00144     account_number = gsb_gui_navigation_get_current_account ();
00145 
00146     /* get the position of the 1st method */

00153     position_1 = g_slist_index ( gsb_data_account_get_sort_list (account_number),
00154                                  GINT_TO_POINTER (payment_number_1));

00159          gsb_data_payment_get_sign (payment_number_2) == GSB_PAYMENT_NEUTRAL

00161          gsb_data_transaction_get_amount (transaction_number_2).mantissa < 0 )
00162         payment_number_2 = -payment_number_2;

00164                                  GINT_TO_POINTER (payment_number_2));
00165     return_value = position_1 - position_2;
00166 
00167     if (!return_value)

00170         const GDate *date_1;

00172

00174         date_2 = gsb_data_transaction_get_date (transaction_number_2);
00175 
00176         /* should not happen */
00177         if ( !date_1 )
00178             return 1;
00179         if ( !date_2 )
00180             return -1;
00181 
00182         return_value = g_date_compare ( date_1,
00183                                         date_2);
00184 
00185         /* no difference in the dates, sort by number of transaction */
