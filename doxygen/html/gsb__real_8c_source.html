00361 gsb_real gsb_real_raw_get_from_string ( const gchar *string,
00362                                         const gchar *mon_thousands_sep,
00363                                         const gchar *mon_decimal_point )
00364 {
00365     static gchar *space_chars;
00366     static gchar *decimal_chars;
00367     static const gchar *positive_chars = "+";
00368     static const gchar *negative_chars = "-";
00369     static const gchar *decimal_char_dot = ".";
00370     static const gchar *decimal_char_comma = ",";
00371     static const gchar *empty_char = "" ;
00372     const gchar *default_decimal_char_dot = decimal_char_dot;
00373     const gchar *default_decimal_char_comma = decimal_char_comma;
00374     unsigned mts_len;
00375     unsigned mdp_len;
00376     unsigned nb_digits = 0;
00377     gint64 mantissa = 0;
00378     gint8 sign = 0;
00379     gint8 dot_position = -1;
00380     const gchar *p = string;
00381     gboolean success = FALSE;
00382 
00383     if ( !string)
00384         return error_real;
00385 
00386     mts_len = mon_thousands_sep
00387               ? strlen ( mon_thousands_sep )
00388               : 0;
00389     mdp_len = mon_decimal_point ? strlen ( mon_decimal_point ) : 0;
00390 
00391     if ( mon_thousands_sep )
00392     {
00393         if ( g_strstr_len ( mon_thousands_sep, -1, decimal_char_dot ) )
00394             default_decimal_char_dot = empty_char;
00395         if ( g_strstr_len ( mon_thousands_sep, -1, decimal_char_comma ) )
00396             default_decimal_char_comma = empty_char ;
00397     }
00398 
00399     decimal_chars = g_strconcat(default_decimal_char_dot,
00400                                 default_decimal_char_comma,
00401                                 mon_decimal_point,
00402                                 NULL);
00403     space_chars = g_strconcat(" ", mon_thousands_sep, NULL);
00404 
00405     for ( ; ; )
00406     {
00407         if ( g_ascii_isdigit ( *p ) )
00408         {
00409             mantissa *= 10;
00410             mantissa += ( *p - '0' );
00411             if ( mantissa > G_MAXLONG )
00412             {
00413                 break;
00414             }
00415             if ( sign == 0 ) sign = 1; /* no sign found yet ==> positive */
00416             ++nb_digits;
00417             ++p;
00418         }
00419         else if ( *p == 0 ) /* terminal zero */
00420         {
00421             success = TRUE;
00422             break;
00423         }
00424         else if ( decimal_chars && strchr ( decimal_chars, *p ) )
00425         {
00426             if ( dot_position >= 0 ) /* already found a decimal separator */
00427             {
00428                 break;
00429             }
00430             dot_position = nb_digits;
00431             p = g_utf8_find_next_char ( p, NULL );
00432         }
00433         else if ( g_utf8_strchr ( space_chars, -1,  g_utf8_get_char( p ) ) )
00434         {
00435             /* just skip spaces and thousands separators */
00436             p = g_utf8_find_next_char ( p, NULL );
00437         }
00438         else if ( strchr ( negative_chars, *p ) )
00439         {
00440             if ( sign != 0 ) /* sign already set */
00441             {
00442                 break;
00443             }
00444             sign = -1;
00445             ++p;
00446         }
00447         else if ( strchr ( positive_chars, *p ) )
00448         {
00449             if ( sign != 0 ) /* sign already set */
00450             {
00451                 break;
00452             }
00453             sign = 1;
00454             ++p;
00455         }
00456         else /* unknown char ==> error */
00457         {
00458             break;
00459         }
00460     }
00461     /* Free memory */
00462     g_free ( decimal_chars );
00463     g_free ( space_chars );
00464     if (success == TRUE)
00465     {
00466         gsb_real result;
00467         result.mantissa = sign * mantissa;
00468         result.exponent = ( dot_position >= 0 )
00469                           ? nb_digits - dot_position
00470                           : 0;
00471         return result;
00472     }
00473     else
00474     {
00475         return error_real;
00476     }
