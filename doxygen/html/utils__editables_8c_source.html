00153         
00154     string = my_strdup (gtk_entry_get_text ( GTK_ENTRY (entry)));

00156     if (string && strlen (string))
00157         pointeur = string + strlen (string);

00162     {
00163         if ( pointeur[0] == '+'

00172     }
