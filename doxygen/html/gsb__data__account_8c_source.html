00224     gint last_number;
00225 
00226     account = g_malloc0 (sizeof ( struct_account ));

00235     /* we have to append the account first because some functions later will

00238

00240     account -> account_number = last_number + 1;

00246     account -> account_kind = account_kind;
00247 
00248     /* select the white line */
00249     account -> current_transaction_number = -1;
00250 
00251     /*     if it's the first account, we set default conf (R not displayed and 3 lines per transaction) */
00252     /*     else we keep the conf of the last account */
00253     /*     same for the form organization
00254      *     same for sorting the transactions list */
00255 
00256     if ( account -> account_number == 1 )
00257     {
00258         account -> nb_rows_by_transaction = 3;
00259 
00260         /* set the form organization by default */
00261         gsb_data_form_new_organization (account -> account_number);
00262         gsb_data_form_set_default_organization (account -> account_number);
00263 
00264         /* sort the transactions by default */
00265         gsb_data_account_set_default_sort_values (account -> account_number);
00266     }
00267     else
00268     {
00269         account -> show_r = gsb_data_account_get_r (last_number);
00270         account -> nb_rows_by_transaction = gsb_data_account_get_nb_rows (last_number);
00271 
00272         /* try to copy the form of the last account, else make a new form */
00273         if ( !gsb_data_form_dup_organization ( last_number,
00274                                                account -> account_number ))
00275         {
00276             gsb_data_form_new_organization (account -> account_number);
00277             gsb_data_form_set_default_organization (account -> account_number);
00278         }
00279 
00280         /* try to copy the sort values of the last account, else set default values */
00281         if ( !gsb_data_form_dup_sort_values ( last_number,
00282                                               account -> account_number ))
00283             gsb_data_account_set_default_sort_values (account -> account_number);
00284     }
00285
