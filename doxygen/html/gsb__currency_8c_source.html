00415 void gsb_currency_check_for_change ( gint transaction_number )
00416 {
00417     gint transaction_currency_number;
00418     gint account_currency_number;
00419     gint link_number;
00420 
00421     account_currency_number = gsb_data_account_get_currency (

00424                         transaction_number );
00425     link_number = gsb_data_currency_link_search ( account_currency_number,
00426                         transaction_currency_number );
00427 
00428     if ( link_number )

00430         if ( current_exchange_fees.mantissa )
00431             gsb_data_transaction_set_exchange_fees ( transaction_number,
00432                         current_exchange_fees );
00433         else
00434             gsb_data_transaction_set_exchange_fees ( transaction_number,
00435                         null_real );
00436 
00437         if ( current_exchange.mantissa == 0 )
00438             gsb_data_transaction_set_exchange_rate ( transaction_number,
00439                     gsb_real_abs (
00440                     gsb_data_currency_link_get_change_rate (
00441                     link_number ) ) );
00442         else
00443             gsb_data_transaction_set_exchange_rate ( transaction_number,
00444                     current_exchange );
00445 
00446         if ( gsb_data_currency_link_get_first_currency (
00447          link_number) == account_currency_number )
00448             gsb_data_transaction_set_change_between (transaction_number, 1 );
00449         else
00450             gsb_data_transaction_set_change_between (transaction_number, 0 );
00451 
00452         return;

00454 
00455     if ( current_exchange.mantissa == 0 )
00456         gsb_currency_exchange_dialog ( account_currency_number,
00457                         transaction_currency_number,
00458                         0,
00459                         null_real,
00460                         null_real,
00461                         TRUE );
00462 
00463     gsb_data_transaction_set_exchange_rate ( transaction_number,
00464                         gsb_real_abs (current_exchange));
00465     if ( current_exchange_fees.mantissa )
00466         gsb_data_transaction_set_exchange_fees ( transaction_number,
00467                         current_exchange_fees );
00468     else
00469         gsb_data_transaction_set_exchange_fees ( transaction_number,
00470                         null_real );
