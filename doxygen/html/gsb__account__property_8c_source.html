01389     if ( g_strcmp0 ( s_iban -> locale, "XX" ) != 0 &&
01390                     g_utf8_strlen ( model, -1 ) != g_utf8_strlen ( tmp_str, -1 ) )

01400      * Ensuite on remplace les lettres par des chiffres selon un
01401      * codage particulier A = 10 B = 11 etc...

01415                 buffer = g_strdup_printf ( "%c", ptr[i] );
01416             else
01417                 buffer = g_strconcat ( buffer,
01418                         g_strdup_printf ("%c", ptr[i]), NULL );
01419         }
