00723 void gsb_data_payee_add_transaction_to_payee ( gint transaction_number )
00724 {
00725     struct_payee *payee;
00726 
00727     payee = gsb_data_payee_get_structure ( gsb_data_transaction_get_party_number (transaction_number));
00728 
00729     /* if no payee in that transaction, and it's neither a split, neither a transfer,
00730      * we work with empty_payee */
00731 
00732     /* should not happen, this is if the transaction has a payee wich doesn't exists
00733      * we show a debug warning and get without payee */
00734     if (!payee)
00735     {
00736         gchar *tmpstr;
00737         tmpstr = g_strdup_printf ( _("The transaction %d has a payee n∞%d but it doesn't exist."),
00738                                    transaction_number, 
00739                                    gsb_data_transaction_get_party_number (transaction_number));
00740         warning_debug (tmpstr);
00741         g_free (tmpstr);
00742         payee = empty_payee;
00743     }
00744 
00745     payee -> payee_nb_transactions ++;
00746     if ( ! gsb_data_transaction_get_split_of_transaction ( transaction_number ) )
00747         payee -> payee_balance = gsb_real_add ( payee -> payee_balance,
00748                                             gsb_data_transaction_get_adjusted_amount_for_currency 
00749                         ( transaction_number, payee_tree_currency (), -1));
