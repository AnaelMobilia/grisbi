00428                 payment_list = payment_list -> next;

00459 
00460             /* Insert a child node */

00462             gtk_tree_store_set (GTK_TREE_STORE (model), &method_iter,
00463                                 PAYMENT_METHODS_NAME_COLUMN, gsb_data_payment_get_name (payment_number),
