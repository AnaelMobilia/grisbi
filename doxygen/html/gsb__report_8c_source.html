00159 gboolean gsb_report_update_report_list ( void )
00160 {
00161     GSList *list_tmp;
00162     GtkTreeIter iter;
00163 
00164     /* if no filter, thats because not created, but don't create here
00165      * because we can come here without needed of report button */
00166     if (!report_model_filter)
00167         return FALSE;
00168 
00169     gtk_list_store_clear (GTK_LIST_STORE (report_model));
00170 
00171     /* fill the list */
00172     list_tmp = gsb_data_report_get_report_list ();
00173 
00174     while ( list_tmp )
00175     {
00176         gint report_number;
00177 
00178         report_number = gsb_data_report_get_report_number (list_tmp -> data);
00179 
00180         gtk_list_store_append ( GTK_LIST_STORE (report_model),
00181                                 &iter );
00182         /* for now REPORT_COL_VIEW is always TRUE, perhaps can change one day */
00183         gtk_list_store_set ( GTK_LIST_STORE (report_model),
00184                              &iter,
00185                              REPORT_COL_NAME, gsb_data_report_get_report_name (report_number),
00186                              REPORT_COL_NUMBER, report_number,
00187                              REPORT_COL_VIEW, TRUE,
00188                              -1 );
00189         list_tmp = list_tmp -> next;
00190     }
00191     return FALSE;
00192 }

00203 gboolean gsb_report_create_combobox_store ( void )
00204 {
00205     /* the report list store, contains 3 columns :
00206      * REPORT_COL_NAME : the name of the report
00207      * REPORT_COL_NUMBER : the number of the report
00208      * REPORT_COL_VIEW : it that report should be showed */
00209 
00210     report_model = GTK_TREE_MODEL ( gtk_list_store_new ( 3,
00211                                                          G_TYPE_STRING,
00212                                                          G_TYPE_INT,
00213                                                          G_TYPE_BOOLEAN ));
00214     report_model_filter = gtk_tree_model_filter_new ( report_model,
00215                                                       NULL );
00216     gtk_tree_model_filter_set_visible_column ( GTK_TREE_MODEL_FILTER (report_model_filter),
00217                                                REPORT_COL_VIEW );
00218     gsb_report_update_report_list ();
00219     return TRUE;
00220 }
