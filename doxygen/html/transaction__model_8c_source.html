00147                                                   gint transaction_number,
00148                                                   gint line_in_transaction )
00149 {
00150     gint i;
00151     gint mother_transaction_number = transaction_number;
00152     gboolean is_child;
00153     CustomRecord *record = NULL;
00154 
00155     g_return_val_if_fail (iter != NULL, FALSE);
00156     g_return_val_if_fail (transaction_number != 0, FALSE);
00157 
00158     /* the transaction can be a mother or a child,
00159      * in all cases, we have to find first the mother (or herself)
00160      * because the children are saved into the mother structure */
00161 
00162     /* if it's a child, look for the mother first */
00163     is_child = gsb_data_transaction_get_mother_transaction_number (transaction_number) != 0;
00164     if (is_child)
00165         mother_transaction_number = gsb_data_transaction_get_mother_transaction_number (transaction_number);
00166 
00167     /* search the mother */
00168     for (i=0 ; i<custom_list -> num_rows ; i++)
00169     {
00170         record = custom_list -> rows[i];
00171 
00172         /* if we are on the good transaction, check the line, or if child can only be that line */
00173         if (record -> what_is_line == IS_TRANSACTION
00174             &&
00175             gsb_data_transaction_get_transaction_number (record -> transaction_pointer) == mother_transaction_number )
00176         {
00177             /* ok, found the mother ; if search a child, stop here and continue to search,
00178              * if want the mother, need to check if it's the good line */
00179             if (is_child)
00180                 break;
00181 
00182             if (record -> line_in_transaction == line_in_transaction)
00183             {
00184                 /* fill the iter */
00185                 iter->stamp     = custom_list->stamp;
00186                 iter->user_data = record;
00187                 return TRUE;
00188             }
00189         }
00190     }
00191 
00192     /* come here only if it's a child, else, iter not found */
00193     if (!is_child)
00194         return FALSE;
00195 
00196     /* for now, record is the mother, we search the child */
00197     for (i=0 ; i < record -> number_of_children ; i++)
00198     {
00199         CustomRecord *child_record;
00200 
00201         child_record = record -> children_rows[i];
00202 
00203         if (child_record -> what_is_line == IS_TRANSACTION
00204             &&
00205             gsb_data_transaction_get_transaction_number (child_record -> transaction_pointer) == transaction_number)
00206         {
00207             /* ok, the child is found ; and only 1 line, so no more check */
00208             /* fill the iter */
00209             iter->stamp     = custom_list->stamp;
00210             iter->user_data = child_record;
00211             return TRUE;
00212         }
00213     }
00214     return FALSE;
00215 }

00319         iter->user_data = custom_list -> rows[record -> pos + 1];
