00274 /* get the line af the file,
00275  * convert it in UTF8 and fill string with that line
00276  *
00277  * \param fichier
00278  * \param string
00279  * \param coding_system the orig coding system of the string
00280  *
00281  * \return EOF, 1 if ok, 0 if problem
00282  * */
00283 gint get_utf8_line_from_file ( FILE *fichier,
00284                         gchar **string,
00285                         const gchar *coding_system )
00286 {
00287     gchar c = 0;
00288     gint i = 0;
00289     gint j = 0;
00290     gchar *pointeur_char = NULL;
00291     gchar *tmp_string;
00292 
00293     if ( !fichier )
00294         return 0;
00295             
00296     /* allocate 30 characters, and increase it 30 by 30 */
00297     pointeur_char = (gchar*)g_realloc(pointeur_char,30*sizeof(gchar));
00298 
00299     if ( !pointeur_char )
00300     {
00301         /* ouch, not enough memory */
00302         dialogue_error ( _("Memory allocation error" ));
00303         return 0;
00304     }
00305 
00306     /* get the string untill \n or \r (windows format) */
00307     c =(gchar)fgetc(fichier);
00308     while ( ( c != '\n' ) && (c != '\r') && !feof (fichier))
00309     {
00310         pointeur_char[j++] = c;
00311 
00312         if ( ++i == 29 )
00313         {
00314             pointeur_char = (gchar*)g_realloc(pointeur_char, j + 1 + 30*sizeof(gchar));
00315 
00316             if ( !pointeur_char )
00317             {
00318                 /* ouch, not enough memory */
00319                 dialogue_error ( _("Memory allocation error" ));
00320                 return 0;
00321             }
00322             i = 0;
00323         }
00324         c =(gchar)fgetc(fichier);
00325     }
00326     pointeur_char[j] = 0;
00327 
00328     /* if we finished on \r, jump the \n after it */
00329     if ( c == '\r' )
00330     {
00331         c =(gchar)fgetc(fichier);
00332         if ( c != '\n' )
00333         {
00334             ungetc ( c, fichier );
00335         }
00336     }
00337 
00338     tmp_string = g_convert ( pointeur_char, -1, "UTF-8", 
00339                              coding_system, NULL, NULL,
00340                              NULL );
00341     if (!tmp_string)
00342     {
00343         devel_debug ("convert to utf8 failed, will use latin2utf8");
00344         tmp_string = latin2utf8 (pointeur_char);
00345         if ( tmp_string == NULL )
00346         {
00347             dialogue_special ( GTK_MESSAGE_ERROR, make_hint (
00348                             _("Convert to utf8 failed."),
00349                             _("If the result is not correct, try again by selecting the "
00350                             "correct character set in the window for selecting files.") ) );
00351             return 0;
00352         }
00353     }
00354     *string = tmp_string;
00355     g_free (pointeur_char);
00356 
00357     if ( feof(fichier))
00358         return EOF;
00359     else
00360         return 1;
00361 }

00527 GSList *utils_files_check_UTF8_validity ( const gchar *contents,
00528                         const gchar *coding_system )
00529 {
00530     GSList *list = NULL;
00531     struct struc_check_encoding *result;
00532     gchar *string = NULL;
00533     gint long_str = 0;
00534     gsize size = 0;
00535     gsize bytes_written = 0;
00536     GError * error = NULL;
00537     gint i = 0;
00538     gchar *ptr;
00539 
00540     ptr = (gchar *) contents;
00541     
00542     while ( strlen ( ptr ) > 0 )
00543     {
00544         gchar *ptr_tmp;
00545 
00546         ptr_tmp = g_strstr_len ( ptr, strlen ( ptr ), "\n" );
00547         if ( ptr_tmp )
00548         {
00549             gchar *ptr_r;
00550 
00551             string = g_strndup ( ptr, ( ( ptr_tmp ) - ptr ) );
00552             if ( ( ptr_r = g_strrstr ( string, "\r" ) ) )
00553                 ptr_r[0] = '\0';
00554 
00555             if ( g_convert ( string, -1, "UTF-8", coding_system, NULL, NULL, NULL ) == NULL )
00556             {
00557                 gchar *tmp_str;
00558 
00559                 long_str = strlen ( string );
00560                 result = g_malloc0 ( sizeof ( struct struc_check_encoding ) );
00561                 result -> charset = "";
00562                 result -> result = string;
00563                 list = g_slist_append ( list, result );
00564                 do
00565                 {
00566                      tmp_str = g_convert ( string, long_str, "UTF-8", charset_array[i],
00567                                 &size, &bytes_written, &error );
00568                     if ( tmp_str )
00569                     {
00570                         result = g_malloc0 ( sizeof ( struct struc_check_encoding ) );
00571                         result -> charset = charset_array[i];
00572                         result -> result = tmp_str;
00573                         list = g_slist_append ( list, result );
00574                     }
00575                     i++;
00576                 } while ( charset_array[i] );
00577 
00578                 return list;
00579             }
00580             g_free ( string );
00581             ptr = ptr_tmp + 1;
00582         }
00583         else
00584             break;
00585     }
