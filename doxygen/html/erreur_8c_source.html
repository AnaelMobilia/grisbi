00182     /* un int pour stocker le level de debug et une chaine qui contient sa version texte */
00183     gint debug_variable=0;
00184     gchar *debug_level="";
00185         gchar* tmpstr1;
00186         gchar* tmpstr2;
00187 
00188     if (getenv ("DEBUG_GRISBI"))
00189     {
00190         /* on choppe la variable d'environnement */
00191         debug_variable=utils_str_atoi (getenv ("DEBUG_GRISBI"));
00192 
00193         /* on verifie que la variable est cohérente */
00194         if (debug_variable > 0 && debug_variable <= MAX_DEBUG_LEVEL) 
00195         {
00196             /* on renseigne le texte du level de debug */
00197             debugging_grisbi = debug_variable;
00198             switch(debug_variable)
00199             {
00200                 case DEBUG_LEVEL_ALERT: { debug_level="Alert"; break; }
00201                 case DEBUG_LEVEL_IMPORTANT: { debug_level="Important"; break; }
00202                 case DEBUG_LEVEL_NOTICE: { debug_level="Notice"; break; }
00203                 case DEBUG_LEVEL_INFO: { debug_level="Info"; break; }
00204                 case DEBUG_LEVEL_DEBUG: { debug_level="Debug"; break; }
00205             }
00206 
00207             /* on affiche un message de debug pour indiquer que le debug est actif */
00208             tmpstr1 = g_strdup_printf(_("GRISBI %s Debug"),VERSION);
00209             tmpstr2 = g_strdup_printf(_("Debug enabled, level is '%s'"),debug_level);
00210             debug_message_string ( tmpstr1 , 
00211                                    __FILE__, __LINE__, __PRETTY_FUNCTION__,
00212                                    tmpstr2,
00213                                    DEBUG_LEVEL_INFO, TRUE);
00214             g_free ( tmpstr1 );
00215             g_free ( tmpstr2 );
00216         }
00217         else
00218         {
00219             /* on affiche un message de debug pour indiquer que le debug est actif */
00220             gchar* tmpstr = g_strdup_printf(_("GRISBI %s Debug"),VERSION);

00222                                   __FILE__, __LINE__, __PRETTY_FUNCTION__,
00223                                   _("Wrong debug level, please check DEBUG_GRISBI environnement variable"),
00224                                   DEBUG_LEVEL_INFO, TRUE);
00225             g_free ( tmpstr );
