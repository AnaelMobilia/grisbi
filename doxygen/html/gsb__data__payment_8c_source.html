00698 gint gsb_data_payment_create_default  ( gint account_number )
00699 {
00700     gint payment_number;
00701     GSList *sort_list = NULL;
00702 devel_debug_int ( account_number);
00703     switch (gsb_data_account_get_kind (account_number))
00704     {
00705         case GSB_TYPE_CASH:
00706             break;
00707 
00708         case GSB_TYPE_ASSET:
00709             break;
00710 
00711         case GSB_TYPE_BANK:
