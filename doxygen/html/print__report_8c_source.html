00142     /* get the width of each columns */
00143     if (columns_width)

00146

00148     for (i=0 ; i<GTK_TABLE (table_etat) -> ncols ; i++)
00149         table_size = table_size + (GTK_TABLE (table_etat) -> cols)[i].allocation;
00150     for (i=0 ; i<GTK_TABLE (table_etat) -> ncols ; i++)
00151         columns_width[i] = (page_width * (GTK_TABLE (table_etat) -> cols)[i].allocation) / table_size;
00152 
00153     /* calculate the nb of rows in 1 page and in the first page */
00154     nb_rows_per_page = page_height / size_row;

00186                                   GtkPrintContext *context,

00188                                   gpointer null )
00189 {
00190     GList *children;
00191     gint nb_columns;
00192     gint nb_lines;
00193     gint rows_drawed = 0;
00194     gint previous_line = -1;
00195     gboolean is_title = FALSE;
00196 
00197     devel_debug_int (page);
00198 
00199     nb_columns = GTK_TABLE (table_etat) -> ncols;
00200     nb_lines = GTK_TABLE (table_etat) -> nrows;
00201 
00202     /* children begins with the last GtkTableChild added, so we need to inverse */
00203     if (page)
00204         children = current_child_table;
00205     else
00206     {
00207         children = g_list_last (GTK_TABLE (table_etat) -> children);
00208         is_title = TRUE;
00209     }
00210 
00211     while (children)
00212     {
00213         GtkTableChild *child;
00214         gint line_position;
00215 
00216         child = children -> data;
00217 
00218         /* if at the to of the page, set previous_line to the first line
00219          * else we check if the page is not finished */
00220         if (previous_line == -1)
00221             previous_line = child -> top_attach;
00222         else
00223             if (child -> top_attach != previous_line)
00224             {
00225                 if ((rows_drawed == nb_rows_per_page && page)
00226                     ||
00227                     (rows_drawed == nb_rows_first_page && !page))
00228                 {
00229                     current_child_table = children;
00230                     current_child_line = child -> top_attach;
00231                     break;
00232                 }
00233                 rows_drawed++;
00234                 previous_line = child -> top_attach;
00235             }
00236 
00237         /* calculate the line position, if first page, add the size title except for the title itself */
00238         line_position = (child -> top_attach - current_child_line) * size_row + (!page * !is_title * size_title);
00239 
00240         if (GTK_IS_LABEL (child -> widget) || GTK_IS_EVENT_BOX (child -> widget))
00241             /* we are on a label, draw the text */
00242             print_report_draw_row (context, child, line_position, is_title);
00243         else
00244         {
00245             /* we are on a separator, draw it */
00246             if (GTK_IS_VSEPARATOR (child->widget))
00247                 print_report_draw_column (child, line_position);
00248             else
00249                 print_report_draw_line (child, line_position);
00250         }
00251         if (is_title)
00252             is_title = FALSE;
00253 
00254         children = children -> prev;
00255     }
00256     return FALSE;
00257 }
