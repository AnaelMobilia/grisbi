00266 gchar* file_selection_get_last_directory(GtkFileChooser* filesel,gboolean ended) 
00267 {/* {{{ */
00268     gchar * dirstr = gtk_file_chooser_get_current_folder ( filesel );
00269     gint     dirstr_len  = strlen(dirstr);
00270     gchar*   sepstr      = my_strdup(G_DIR_SEPARATOR_S);
00271     gint     sepstr_len  = strlen(sepstr);
00272     gboolean is_endedstr = FALSE;
00273     gchar* tmpstr;
00274 
00275     /* Chek if the sirectory string is ended by a separator
00276      (if directory string  is small than the separator string
00277      it can ot be ended by the separator string) */
00278     if ( dirstr_len >= sepstr_len) 
00279     {
00280         is_endedstr = (gboolean)(!strncmp( dirstr + dirstr_len - sepstr_len, sepstr, sepstr_len));
00281     }
00282 
00283     /* We want the dirstr ended by a sepstrarator but there no */
00284     if (ended&&!is_endedstr)
00285     {
00286         tmpstr = g_strconcat(dirstr,G_DIR_SEPARATOR_S,NULL);
00287         g_free(dirstr);
00288         dirstr = tmpstr;
00289     }
00290     /* We do not want to have a separator at the end, but there is one */
00291     else if ((!ended)&&is_endedstr)
00292     {
00293         dirstr[dirstr_len-sepstr_len-1] = 0;
00294     }
00295     
00296     return g_filename_to_utf8(dirstr,-1,NULL,NULL,NULL);
00297 
00298 } /* }}} file_selection_get_last_directory */
