01763 void gsb_form_check_auto_separator ( GtkWidget *entry )
01764 {
01765     gint account_number;
01766     gchar *string;
01767     gint floating_point;
01768     gchar *tmp = NULL;
01769     gint i;
01770 
01771     if (!etat.automatic_separator
01772         ||
01773         !entry )
01774         return;
01775     
01776     /* we need a my_strdup to permit to do the g_free later
01777      * because if strlen < floating point we need to
01778      * malloc another string */
01779     string = my_strdup (gtk_entry_get_text (GTK_ENTRY(entry)));
01780     if ( !string || strlen (string) == 0 )
01781         return;
01782 
01783     account_number = gsb_form_get_account_number ();
01784     floating_point = gsb_data_currency_get_floating_point (gsb_data_account_get_currency (account_number));
01785 
01786     if ( strchr (string, '.')
01787          ||
01788          strchr (string, ','))
01789     {
01790         g_free (string);
01791         return;
01792     }
01793 
01794     /* if string is < the floating_point, increase it to have
01795      * 1 character more (to set the 0 before the .) */
01796     if (strlen(string) <= floating_point)
01797     {
01798         gchar *concat_tmp;
01799 
01800         tmp = g_malloc (floating_point - strlen(string) + 2);
01801         for (i=0 ; i<(floating_point - strlen(string) + 1) ; i++)
01802             tmp[i] = '0';
01803         tmp[floating_point - strlen(string) + 1] = 0;
01804         concat_tmp = g_strconcat ( tmp,
01805                                    string,
01806                                    NULL );
01807         g_free (tmp);
01808         g_free (string);
01809         string = concat_tmp;
01810     }
01811 
01812     tmp = g_malloc ((strlen(string)+2) * sizeof (gchar));
01813 
01814     memcpy ( tmp, string, strlen(string) - floating_point);
01815 
01816     i = strlen(string) - floating_point;
01817     tmp[i] = '.';
01818     i++;
01819     memcpy ( tmp + i,
01820              string + i - 1,
01821              floating_point );
01822     i = i + floating_point;
01823     tmp[i] = 0;
01824     gtk_entry_set_text (GTK_ENTRY (entry),
01825                         tmp );
01826     g_free (tmp);
01827     g_free (string);
