00219     GSList *list_tmp;

00230         GSList *sorted_list;
00231         GSList *payment_list;
00232         gboolean visible;
00233 
00234         account_number = gsb_data_account_get_no_account ( list_tmp -> data );
00235 
00236         /* when no method of payment, hide the checkbuttons because non sense */
00237         payment_list = gsb_data_payment_get_list_for_account (account_number);
00238         if (payment_list)

00249                             RECONCILIATION_SORT_VISIBLE_COLUMN, visible,
00250                             RECONCILIATION_SORT_SORT_COLUMN, gsb_data_account_get_reconcile_sort_type(account_number),

00254                             RECONCILIATION_SORT_SENSITIVE_COLUMN, visible,
00255                             -1 );
00256 
00257         /* the sorted list is a list of numbers of method of payment, in the good order */
00258         sorted_list = gsb_data_account_get_sort_list (account_number);
00259 
00260         while ( sorted_list )
00261         {
00262             gint payment_number;
00263 
00264             /* in the sorted list, a number can be negative, when split a neutral into 2 parts
00265              * so payment number here can be negative for neutral payments */
00266             payment_number = GPOINTER_TO_INT (sorted_list -> data);
00267 
00268             if (payment_number)
00269             {
00270                 gchar *name = NULL;
00271 
00272                 gtk_tree_store_append (GTK_TREE_STORE (model),
00273                                        &payment_method_iter, 
00274                                        &account_iter);
00275 
00276                 /* if split the neutrals, show here with the name */
00277                 switch (gsb_data_payment_get_sign (abs (payment_number)))
00278                 {
00279                     case GSB_PAYMENT_DEBIT:
00280                     case GSB_PAYMENT_CREDIT:
00281                         name = my_strdup (gsb_data_payment_get_name (payment_number));
00282                         break;
00283 
00284                     case GSB_PAYMENT_NEUTRAL:
00285                         if (gsb_data_account_get_split_neutral_payment (account_number))
00286                         {
00287                             /* the neutrals are splitted */
00288                             if (payment_number < 0)
00289                                 name = g_strconcat ( gsb_data_payment_get_name (-payment_number),
00290                                                      " ( - )", NULL );
00291                             else

00293                                                      " ( + )", NULL );
00294                         }
00295                         else
00296                             name = my_strdup (gsb_data_payment_get_name (payment_number));
00297                         break;

00299 
00300                 gtk_tree_store_set (GTK_TREE_STORE (model), &payment_method_iter,

00303                                     RECONCILIATION_SORT_SORT_COLUMN, FALSE,

00305                                     RECONCILIATION_SORT_ACCOUNT_COLUMN, account_number,
