00104     /* First, we iterate over all locales except for C to avoid mixing

00107     while ( *languages )
00108     {
00109         GSList * list;

00115                                                            model );

00117             {
00118                 if ( ! g_slist_find_custom ( category_files, list -> data, 
00119                                              (GCompareFunc) cherche_string_equivalente_dans_slist ) )
00120                 {
00121                     category_files = g_slist_append ( category_files, list -> data );
00122                 }
00123                 list = list -> next;
00124             }
00125         }
00126 
00127         languages++;
00128     }
00129     if ( ! category_files )

00131         category_files = gsb_category_assistant_scan_directory ( "C", model );

00136     while ( category_files )
00137     {
00138         gsb_category_assistant_parse_file ( (gchar * ) category_files -> data, 
00139                                             model );

00396                                             const gchar ** attribute_names, const gchar ** attribute_values,
