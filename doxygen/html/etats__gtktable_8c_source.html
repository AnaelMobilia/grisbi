00283 {
00284     gint archive_number;

00286 
00287     account_number = gsb_data_transaction_get_account_number (transaction_number);
00288 
00289     /* if it's an archived transaction, open the archive */
00290     archive_number = gsb_data_transaction_get_archive_number (transaction_number);
00291     if (archive_number)
00292     {
00293         /* re-filter the tree view because if we go directly into the report
00294          * and the model was never filtered, we have a nice crash */
00295         transaction_list_filter (account_number);
00296         gsb_transactions_list_restore_archive (archive_number, FALSE);
00297     }
00298 
00299     if ( transaction_number )
00300     {
00301         gint mother_transaction;
00302 
00303         /* go on the good account */
00304         gsb_gui_navigation_set_selection ( GSB_ACCOUNT_PAGE, 
00305                                            account_number,
00306                                            GINT_TO_POINTER (-1));
00307 
00308         /* récupération de la ligne de l'opé dans la liste ; affichage de toutes les opé si nécessaire */
00309         if ( gsb_data_transaction_get_marked_transaction (transaction_number) == OPERATION_RAPPROCHEE 
00310              &&
00311              !gsb_data_account_get_r ( account_number ) )
00312     {
00313         gsb_data_account_set_r ( account_number, TRUE );
00314         gsb_menu_update_view_menu ( account_number );
00315         mise_a_jour_affichage_r ( TRUE );
00316     }
00317         
00318         /* if it's a child, open the mother */
00319         mother_transaction = gsb_data_transaction_get_mother_transaction_number (transaction_number);
00320         if (mother_transaction)
00321             gsb_transactions_list_switch_expander (mother_transaction);
