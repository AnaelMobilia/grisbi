00188

00190                *tmp == ',')
00191         {
00192             *tmp=0;
00193             tmp--;
00194         }
00195     }
00196     else
00197     {
00198         font_description = NULL;
00199         font_name = my_strdup (_("No font defined"));
00200         font_size = NULL;
00201     }
00202 
00203     gtk_widget_modify_font (font_name_label, font_description);
00204 
00205     gtk_label_set_text (GTK_LABEL(font_name_label), font_name);
00206     gtk_label_set_text (GTK_LABEL(font_size_label), font_size);
