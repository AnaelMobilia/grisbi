00306 
00307     last_name = my_strdup (gsb_data_reconcile_get_name (reconcile_number));
00308 
00309     if (last_name)
00310     {
00311         gchar *tmp_pointer;
00312         gchar *end_pointer;
00313 
00314         /* we try to find some digits at the end of the name,
00315          * if found, get the biggest number untill we find a non digit character */
00316         end_pointer = last_name + (strlen ( last_name ) - 1) * sizeof (gchar);
00317         tmp_pointer = end_pointer;
00318 
00319         while ( isdigit ( tmp_pointer[0] ) && tmp_pointer >= last_name )
00320             tmp_pointer--;
00321 
00322         if ( tmp_pointer != end_pointer )
00323         {
00324             /* ok we have some digits at the end of the name */
00325             gchar *zero_string;
00326             gint digit_size;
00327             gint new_digit_size;
00328             gchar *new_string;
00329             gchar *digit_string;
00330                 gchar* oldstr;
00331 
00332             /* tmp_pointer is on the first non digit from the end of the last_name,
00333              * so go to the first digit */
00334             tmp_pointer++;
00335 
00336             digit_string = my_strdup ( tmp_pointer );
00337             tmp_pointer[0] = 0;
00338 
00339             /* increase the number */
00340             digit_size = strlen ( digit_string );
00341             oldstr =  digit_string;
00342             digit_string = utils_str_itoa ( utils_str_atoi ( digit_string ) + 1 );
00343             g_free ( oldstr );
00344             new_digit_size = strlen ( digit_string );
00345 
00346             /* if new_digit_size is < of digit_size, it's because some 0 diseappear
00347              * while the atoi and utils_str_itoa, so we set again that 0
00348              * ie if we had 0007, we want 0008 and no 8 */
00349             if ( new_digit_size < digit_size )
00350             {
00351                 gint i;
00352 
00353                 zero_string = g_malloc ((digit_size-new_digit_size+1)*sizeof (gchar));
00354 
00355                 for ( i=0 ; i<digit_size-new_digit_size ; i++ )
00356                     zero_string[i]='0';
00357 
00358                 zero_string[digit_size-new_digit_size] = 0;
00359             }
00360             else
00361                 zero_string = g_strdup ("");
00362 
00363             /* create the new string */
00364             new_string = g_strconcat ( last_name,
00365                                        zero_string,
00366                                        digit_string,
00367                                        NULL );
00368             g_free (last_name);
00369             g_free (zero_string);
00370             g_free (digit_string);
00371             last_name = new_string;
00372         }

00374                              last_name );
