00182     {
00183         modification_fichier ( FALSE );

00197             gsb_data_scheduled_remove_scheduled (scheduled_number);

00201

00204     list_tmp = gsb_data_transaction_get_complete_transactions_list ();

00210

00215         {
00216             gint contra_transaction_number;
00217

00226             /* now can remove the transaction */
00227             gsb_data_transaction_remove_transaction_without_check ( transaction_number );
00228         }
00229     }
00230 
00231     /* delete the payment_number */
00232     list_tmp = gsb_data_account_get_sort_list ( deleted_account );
00233     while (list_tmp)
