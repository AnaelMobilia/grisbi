00099     CustomRecord *white_record[TRANSACTION_LIST_ROWS_NB];
00100     gint white_line_number;
00101     gint i, j;
00102     gulong newsize;
00103 
00104     custom_list = (CustomList*) g_object_new (CUSTOM_TYPE_LIST, NULL);
00105 
00106     if ( custom_list == NULL )
00107         return NULL;
00108 
00109     /* increase the table of CustomRecord */
00110     custom_list->num_rows = TRANSACTION_LIST_ROWS_NB;
00111     newsize = custom_list->num_rows * sizeof(CustomRecord*);
00112     custom_list->rows = g_malloc0 (newsize);
00113     custom_list->visibles_rows = g_malloc0 (newsize);
00114 
00115     /* create and fill the white line record */
00116     white_line_number = gsb_data_transaction_new_white_line (0);
00117 
00118     /* now we can save the 4 new rows, ie the complete white line */
00119     for (i=0 ; i<TRANSACTION_LIST_ROWS_NB ; i++)
00120     {
00121         /* fill the white_record structure */
00122         white_record[i] = g_malloc0 (sizeof (CustomRecord));
00123         white_record[i] -> transaction_pointer = gsb_data_transaction_get_pointer_of_transaction (white_line_number);
00124         white_record[i] -> what_is_line = IS_TRANSACTION;
00125         white_record[i] -> line_in_transaction = i;
00126 
00127         /* save the newrecord pointer */
00128         custom_list->rows[i] = white_record[i];
00129         /* and the pos (number) of the row */
00130         white_record[i]->pos = i;
00131     }
00132 
00133     /* save the records in all the records of the transaction */
00134     for (i=0 ; i < TRANSACTION_LIST_ROWS_NB ; i++)
00135         for (j=0 ; j < TRANSACTION_LIST_ROWS_NB ; j++)
00136             white_record[i] -> transaction_records[j] = white_record[j];

00389

00391     depth   = gtk_tree_path_get_depth(path);
00392 
00393     mother_n = indices[0];
00394     /* get the first depth record */
00395     if ( mother_n >= custom_list->num_visibles_rows || mother_n < 0 )
00396         return FALSE;
00397 
00398     record = custom_list -> visibles_rows[mother_n];
00399     g_return_val_if_fail (record != NULL, FALSE);
00400 
00401     /* if we want a child, get it here */
00402     if (depth == 2)
00403     {
00404         gint child_n;
00405 
00406         child_n = indices[1];
00407         if (child_n >= record -> number_of_children || child_n < 0)
00408             return FALSE;
00409 
00410         record = record -> children_rows[child_n];
00411         g_return_val_if_fail (record != NULL, FALSE);
00412     }
00413 
00414     /* We simply store a pointer to our custom record in the iter */
00415     iter->stamp      = custom_list->stamp;

00659 static gboolean custom_list_iter_next (GtkTreeModel  *tree_model,
00660                                        GtkTreeIter   *iter)
00661 {
00662     CustomRecord  *record;
00663     CustomList    *custom_list;
00664 
00665     g_return_val_if_fail (CUSTOM_IS_LIST (tree_model), FALSE);
00666     g_return_val_if_fail (iter != NULL, FALSE);
00667     g_return_val_if_fail (iter->user_data != NULL, FALSE);
00668 
00669     custom_list = CUSTOM_LIST(tree_model);
00670 
00671     record = (CustomRecord *) iter->user_data;
00672 
00673     /* 2 way now, either we are on a mother, either on a child */
00674     if (record -> mother_row)
00675     {
00676         /* it's a child iter */
00677         CustomRecord  *mother_record;
00678 
00679         mother_record = record -> mother_row;
00680         if (record -> pos == (mother_record -> number_of_children - 1))
00681             /* we are on the list child, no next */
00682             return FALSE;
00683 
00684         /* go to the next child */
00685         iter->stamp     = custom_list->stamp;
00686         iter->user_data = mother_record -> children_rows[record -> pos + 1];
00687     }
00688     else
00689     {
00690         /* it's a mother iter */
00691         if (record -> filtered_pos == (custom_list -> num_visibles_rows - 1))
00692             /* we are on the last mother, no next */
00693             return FALSE;
00694 
00695         iter->stamp     = custom_list->stamp;
00696         iter->user_data = custom_list -> visibles_rows[record -> filtered_pos + 1];
00697     }
