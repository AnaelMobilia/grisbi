00310                         GtkNotebookPage *page,
00311                         guint page_number,
00312                         gpointer null )
00313 {
00314     gint account_number;
00315 
00316     /* devel_debug_int (page_number); */

00318     {
00319     case GSB_TRANSACTIONS_PAGE:
00320         gsb_form_set_expander_visible ( TRUE, TRUE );
00321         break;
00322     case GSB_ESTIMATE_PAGE:
00323         gsb_form_set_expander_visible (FALSE, FALSE );
00324         account_number = gsb_gui_navigation_get_current_account ( );
00325         if ( gsb_data_account_get_bet_maj ( account_number ) )
00326             bet_data_update_bet_module ( account_number, GSB_ESTIMATE_PAGE );
00327         break;
00328     case GSB_HISTORICAL_PAGE:
00329         gsb_form_set_expander_visible (FALSE, FALSE );
00330         account_number = gsb_gui_navigation_get_current_account ( );
00331         if ( gsb_data_account_get_bet_maj ( account_number ) )
00332             bet_data_update_bet_module ( account_number, GSB_HISTORICAL_PAGE );
00333         bet_historical_set_page_title ( account_number );
00334         break;
00335     case GSB_FINANCE_PAGE:
00336         gsb_form_set_expander_visible (FALSE, FALSE );
00337         account_number = gsb_gui_navigation_get_current_account ( );
00338         bet_finance_ui_update_amortization_tab ( account_number );
00339     case GSB_PROPERTIES_PAGE:
00340         gsb_form_set_expander_visible (FALSE, FALSE );
00341         break;
00342     }
00343 
00344     return ( FALSE );
