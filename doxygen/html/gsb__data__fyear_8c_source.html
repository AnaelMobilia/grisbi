00584 gboolean gsb_data_fyear_check_for_invalid ( gint fyear_number )
00585 {
00586     struct_fyear *fyear;
00587 
00588     fyear = gsb_data_fyear_get_structure ( fyear_number );
00589 
00590     if (!fyear)
00591         return FALSE;
00592 
00593     /* check if there is some good date */
00594     if (!fyear -> beginning_date
00595         ||
00596         !fyear -> end_date)
00597     {
00598         fyear -> invalid_fyear = FYEAR_INVALID_DATE;
00599         return TRUE;
00600     }
00601 
00602     /* first : we check that the first date is above the second */
00603     if ( g_date_compare (fyear -> beginning_date,
00604                          fyear -> end_date) > 0)
00605     {
00606         fyear -> invalid_fyear = FYEAR_INVALID_DATE_ORDER;
00607         return TRUE;
00608     }
00609 
00610     /* second : we check if there is not a cross with another fyear */
00611     if ( (gsb_data_fyear_get_from_date (fyear -> beginning_date) != fyear_number)
00612          ||
00613          (gsb_data_fyear_get_from_date (fyear -> end_date) != fyear_number))
00614     {
00615         fyear -> invalid_fyear = FYEAR_INVALID_CROSS;
00616         return TRUE;
00617     }
00618 
00619     /* it's ok, the fyear is valid */
00620     fyear -> invalid_fyear = FALSE;
00621 
00622     return FALSE;
00623 }

00702         return 0;

00706     if ( ! fyear_2 -> beginning_date )
00707         return -1;
