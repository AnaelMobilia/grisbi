00666              * next visible row */

00670              * either the transaction is visible but that row is not visible

00704                 }

00728                         record -> line_in_transaction); */

00731                 /* if the record was already shown and the position hasn't change, nothing to do */
00732                 if (previous_shown && (record -> filtered_pos == last_pos_filtered_list))
00733                 {
00734                     /* the row itself didn't change but perhaps it got or losed some children
00735                      * in that case, the children were moved before, so we just have to 
00736                      * tell to the tree view that children changed */
00737                     gtk_tree_model_row_has_child_toggled (GTK_TREE_MODEL (custom_list),
00738                                                           path, &iter);
00739                     gtk_tree_path_next (path);
00740                     continue;
00741                 }
00742                 break;
00743         }
00744 
00745         /* now, the record is not at the same place as before (or was not shown before),
00746          * so,
00747          * either we modify the current row if it exits already,
00748          * either we insert a new row if we reached the end */
00749         if (current_pos_filtered_list > previous_visible_rows)
00750             /* we reached the end of list, need to add a row */
00751             gtk_tree_model_row_inserted ( GTK_TREE_MODEL(custom_list),
00752                                           path, &iter);
00753         else
00754             /* ok, the end of list is not reached, we can just change the row */
00755             gtk_tree_model_row_changed ( GTK_TREE_MODEL (custom_list),
00756                                          path, &iter );
00757 
00758         /* the new row can have some children, or the last row (if we just changed the row)
00759          * could have some children we don't want now. that function is magic because
00760          * automatically update if there are children or not */
00761         if (record -> what_is_line == IS_TRANSACTION)
00762             gtk_tree_model_row_has_child_toggled (GTK_TREE_MODEL (custom_list),
00763                                                   path, &iter);
00764 
00765         gtk_tree_path_next (path);
00766     }
00767 
00768     /* if the previous list was bigger than now, we need to delete some rows */
00769     if (previous_visible_rows > current_pos_filtered_list)
00770         for (i=0 ; i < (previous_visible_rows - current_pos_filtered_list) ; i++)
00771             gtk_tree_model_row_deleted ( GTK_TREE_MODEL (custom_list),
00772                                          path );
00773     gtk_tree_path_free(path);
00774 }

01685 static void transaction_list_append_child ( gint transaction_number )
01686 {
01687     gulong newsize;
01688     guint pos;
01689     gint i;
01690     CustomRecord *newrecord;
01691     CustomRecord *mother_record;
01692     GtkTreeIter mother_iter;
01693     CustomRecord **new_children_rows;
01694     gint new_number_of_children;
01695     CustomList *custom_list;
01696     CustomRecord *white_record;
01697 
01698     custom_list = transaction_model_get_model ();
01699 
01700     /* the children are stored into a table of pointer into each mother
01701      * the filter will determine later what mother row will have the
01702      * expander according to the number of row showed */
01703 
01704     /* create the row */
01705     newrecord = transaction_list_create_record (transaction_number, 0);
01706 
01707     /* find 1 of the mother row (the others are into the structure of the mother)
01708      * usually, it's the last record appended */
01709     mother_record = last_mother_appended;
01710 
01711     if ( !mother_record
01712          ||
01713          (gsb_data_transaction_get_transaction_number (mother_record -> transaction_pointer)
01714           !=
01715           gsb_data_transaction_get_mother_transaction_number (transaction_number)))
01716     {
01717         /* the last record appended was not the mother, need to search into the entire list */
01718         if ( transaction_model_get_transaction_iter ( &mother_iter,
01719                         gsb_data_transaction_get_mother_transaction_number (
01720                         transaction_number),
01721                         0 ) )
01722             mother_record = (CustomRecord *) mother_iter.user_data;
01723         else
01724         {
01725             /* we didn't find the mother of the child, we append the child number to
01726              * orphan_child_transactions and the function wich called here has to play
01727              * with it
01728              * (it happens in the versions before 0.6 when the children could have a number
01729              * before the mother */
01730             orphan_child_transactions = g_slist_append ( orphan_child_transactions,
01731                                  GINT_TO_POINTER (transaction_number));
01732             return;
01733         }
01734     }
01735 
01736     /* ok, at this level, mother_record is ok and is one of the row of the mother transaction
01737      * we go on the first mother */
01738     mother_record = mother_record -> transaction_records[0];
01739 
01740     /* set by default the mother row of the child. avoids a crash when creating 
01741      * an archive directly on the home page */
01742     newrecord -> mother_row = mother_record;
01743 
01744     /* get the new number of the row into the mother */
01745     pos = mother_record -> number_of_children;
01746 
01747     /* it must have a child, because while creating the mother, it makes it
01748      * so if no child, better to stop here */
01749     if (!pos)
01750     {
01751         gchar *tmpstr;
01752         tmpstr = g_strdup_printf (
01753                         _("Trying to append the child number %d to the mother %d "
01754                         "in the model, but no white line was created before... "
01755                         "Better to stop here, please contact the Grisbi team to "
01756                         "fix that issue."),
01757                         transaction_number, gsb_data_transaction_get_mother_transaction_number (transaction_number));
01758         dialogue_error (tmpstr);
01759         g_free (tmpstr);
01760         g_free (newrecord);
01761         return;
01762     }
01763 
01764     /* increase the children rows */
01765     new_number_of_children = mother_record -> number_of_children + 1;
01766     newsize = new_number_of_children * sizeof (CustomRecord*);
01767     new_children_rows = g_realloc (mother_record -> children_rows,
01768                                    newsize);
01769 
01770     /* save the new record before the white line */
01771     white_record = new_children_rows[pos -1];
01772 
01773     new_children_rows[pos - 1] = newrecord;
01774     newrecord -> pos = pos - 1;
01775     newrecord -> filtered_pos = pos -1;
01776 
01777     new_children_rows[pos] = white_record;
01778     white_record -> pos = pos;
01779     white_record -> filtered_pos = pos;
01780 
01781     /* set the color */
01782     newrecord -> row_bg = &split_background;
01783 
01784     /* we can now save the new number of children and the new children rows into their mothers */
01785     for (i=0 ; i < TRANSACTION_LIST_ROWS_NB ; i++)
01786     {
01787         mother_record -> transaction_records[i] -> children_rows = new_children_rows;
01788         mother_record -> transaction_records[i] -> number_of_children = new_number_of_children;
01789 
01790         /* set the good mother row of the child */
01791         if (mother_record -> transaction_records[i] -> has_expander)
01792         {
01793             newrecord -> mother_row = mother_record -> transaction_records[i];
01794             white_record -> mother_row = mother_record -> transaction_records[i];
01795         }
01796     }
01797 
01798     /* we need now to recalculate the amount of split and update the white line */
01799     transaction_list_update_white_child (white_record);
01800 
01801     if (mother_record -> filtered_pos != -1)
01802     {
01803         /* the mother is visible, inform the tree view we append a child */
01804         GtkTreePath *path;
01805         GtkTreeIter iter;
01806     gint last_line;
01807 
01808         /* we go on the last mother record, wich contains the expander */
01809     last_line = transaction_list_get_last_line ( custom_list -> nb_rows_by_transaction );
01810         mother_record = mother_record -> transaction_records[last_line];
01811 
01812         newrecord -> mother_row = mother_record;
01813         /* this is very important to keep to compatibility with the normal transactions */
01814         newrecord -> transaction_records[0] = newrecord;
01815 
01816         /* update the new line, wich replace the white line */
01817         path = gtk_tree_path_new();
01818         gtk_tree_path_append_index(path, mother_record->filtered_pos);
01819         gtk_tree_path_append_index(path, newrecord->filtered_pos);
01820 
01821         iter.stamp = custom_list -> stamp;
01822         iter.user_data = newrecord;
01823 
01824         gtk_tree_model_row_changed (GTK_TREE_MODEL(custom_list), path, &iter);
01825         gtk_tree_path_free(path);
01826 
01827         /* update white line, wich moved to the end */
01828         path = gtk_tree_path_new();
01829         gtk_tree_path_append_index(path, mother_record->filtered_pos);
01830         gtk_tree_path_append_index(path, white_record->filtered_pos);
01831 
01832         iter.stamp = custom_list -> stamp;
01833         iter.user_data = white_record;
01834 
01835         gtk_tree_model_row_inserted (GTK_TREE_MODEL(custom_list), path, &iter);
01836         gtk_tree_path_free(path);
01837     }
