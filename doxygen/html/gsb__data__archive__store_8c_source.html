00167 void gsb_data_archive_store_create_list ( void )
00168 {
00169     GSList *tmp_list;
00170 
00171     tmp_list = gsb_data_transaction_get_complete_transactions_list ();
00172     while (tmp_list)
00173     {
00174     gint transaction_number;
00175     gint archive_number;
00176 
00177     transaction_number = gsb_data_transaction_get_transaction_number (tmp_list -> data);
00178 
00179     archive_number = gsb_data_transaction_get_archive_number (transaction_number);
00180 
00181     if (archive_number)
00182     {
00183         struct_store_archive *archive_store;
00184         gint floating_point;
00185         gint account_number;
00186 
00187         account_number = gsb_data_transaction_get_account_number (transaction_number);
00188         floating_point = gsb_data_currency_get_floating_point (
00189                             gsb_data_account_get_currency (account_number) );
00190         archive_store = gsb_data_archive_store_find_struct ( archive_number, account_number);
00191         if (archive_store)
00192         {
00193             /* there is already a struct_store_archive for the same archive and the same account,
00194              * we increase the balance except for operations */
00195             if ( !gsb_data_transaction_get_mother_transaction_number ( transaction_number ) )
00196                 archive_store -> balance = gsb_real_add ( archive_store -> balance,
00197                                     gsb_data_transaction_get_adjusted_amount (
00198                                     transaction_number,
00199                                     floating_point));
00200             archive_store -> nb_transactions++;
00201         }
00202         else
00203         {
00204             /* there is no struct_store_archive for that transaction, we make a new one 
00205              * with the balance of the transaction as balance */
00206             gint archive_store_number;
00207 
00208             archive_store_number = gsb_data_archive_store_new ();
00209             archive_store = gsb_data_archive_store_get_structure (archive_store_number);
00210 
00211             archive_store -> archive_number = archive_number;
00212             archive_store -> account_number = account_number;
00213             archive_store -> balance = gsb_data_transaction_get_adjusted_amount (
00214                             transaction_number, floating_point);
00215             if ( ! gsb_data_transaction_get_mother_transaction_number ( transaction_number ) )
00216                 archive_store -> nb_transactions = 1;
00217         }
00218     }
00219     tmp_list = tmp_list -> next;
00220     }
