00145 static struct_text_comparison *gsb_data_report_text_comparison_get_struct_by_no ( gint text_comparison_number )
00146 {
00147     GSList *text_comparison_list_tmp;
00148 
00149     /* check first if the text_comparison is in the buffer */
00150 
00151     if ( text_comparison_buffer
00152          &&
00153          text_comparison_buffer -> text_comparison_number == text_comparison_number )
00154         return text_comparison_buffer;
00155 
00156     text_comparison_list_tmp = text_comparison_list;
00157 
00158     while ( text_comparison_list_tmp )
00159     {
00160         struct_text_comparison *text_comparison;
00161 
00162         text_comparison = text_comparison_list_tmp -> data;
00163 
00164         if ( text_comparison -> text_comparison_number == text_comparison_number )
00165         {
00166             text_comparison_buffer = text_comparison;
00167             return text_comparison;
00168         }
00169 
00170         text_comparison_list_tmp = text_comparison_list_tmp -> next;
00171     }
00172 
00173     /* here, we didn't find any text_comparison with that number */
00174 
00175     return 0;
00176 }
