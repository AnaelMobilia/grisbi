00501 {
00502     GtkComboFixPrivate *priv = combofix -> priv;
00503     GtkTreeIter iter;
00504     gboolean case_sensitive = priv -> case_sensitive;

00508 
00509     while ( valid )
00510     {
00511         gchar *tmp_str;
00512         gboolean separator;
00513 
00514         gtk_tree_model_get ( GTK_TREE_MODEL ( priv -> store ), &iter,
00515                     COMBOFIX_COL_REAL_STRING, &tmp_str,
00516                     COMBOFIX_COL_SEPARATOR, &separator,
00517                     -1 );
00518 
00519         if ( separator )
00520         {
00521             valid = gtk_tree_model_iter_next ( GTK_TREE_MODEL ( priv -> store ), &iter );
00522             continue;
00523         }

00525         {
00526             g_free ( tmp_str );

00528         }
00529         else if ( !g_utf8_collate ( g_utf8_casefold ( text, -1 ),
00530                                     g_utf8_casefold ( tmp_str, -1 ) ) )
00531         {

01089 static gchar *gtk_combofix_update_visible_rows ( GtkComboFix *combofix,
01090                         const gchar *string )
01091 {
01092     GtkTreeModel *model;
01093     gchar *complete_string = NULL;
01094     GtkTreePath *path;
01095     GtkTreeIter iter;
01096     gint path_ok;
01097     gint length;
01098     gboolean text_written = FALSE;
01099     gboolean separator = FALSE;
01100     GtkComboFixPrivate *priv = combofix -> priv;
01101 
01102     if (!combofix
01103         ||
01104         !string )
01105         return NULL;
01106 
01107     length = strlen (string);
01108     if (!length)
01109         return NULL;
01110 
01111     priv -> visible_items = 0;
01112     model = GTK_TREE_MODEL ( priv -> store );
01113     path = gtk_tree_path_new_first ( );
01114     path_ok = gtk_tree_model_get_iter ( model, &iter, path );
01115 
01116     while ( path_ok )
01117     {
01118         gchar *model_string;
01119         gint show_row = 0;
01120         gint model_string_length;
01121 
01122         gtk_tree_model_get ( model,
01123                         &iter,
01124                         COMBOFIX_COL_REAL_STRING, &model_string,
01125                         COMBOFIX_COL_SEPARATOR, &separator,
01126                         -1 );
01127 
01128         /* The separators are never showed */
01129         if (separator)
01130             show_row = 0;
01131         else
01132         {
01133             model_string_length = strlen ( model_string );
01134 
01135             if ( priv -> case_sensitive )
01136             {
01137                 show_row = !strncmp ( model_string,
01138                             string,
01139                             MIN ( length, model_string_length ) );
01140             }
01141             else
01142                 show_row = !g_strncasecmp ( model_string,
01143                             string,
01144                             MIN ( length, model_string_length ) );
01145 
01146             if ( show_row )
01147             {
01148                 /* if the current checked string is exactly the same as the wanted string,
01149                  * we keep it for completion, else we keep only the first approximation */
01150                 if ( model_string_length == length )
01151                     complete_string = model_string;
01152                 if ( !complete_string && model_string_length > length )
01153                     complete_string = model_string;
01154 
01155                 priv -> visible_items ++;
01156                 text_written = TRUE;
01157             }
01158         }
01159         gtk_tree_store_set ( GTK_TREE_STORE (model),
01160                         &iter,
01161                         COMBOFIX_COL_VISIBLE, show_row,
01162                         -1 );
01163 
01164         /* increment the path :
01165          *      go to see the children only if the mother is showed */
01166         
01167         if ( gtk_tree_model_iter_has_child ( model, &iter )
01168              &&
01169              show_row )
01170             gtk_tree_path_down ( path );
01171         else
01172             gtk_tree_path_next (path);
01173 
01174         path_ok = gtk_tree_model_get_iter ( model, &iter, path );
01175 
01176         /* if path_ok is FALSE, perhaps we are on the end of the children list... */
01177         if ( !path_ok
01178             &&
01179             gtk_tree_path_get_depth ( path ) > 1 )
01180         {
01181             gtk_tree_path_up ( path );
01182             gtk_tree_path_next ( path );
01183             path_ok = gtk_tree_model_get_iter ( model, &iter, path );
01184         }
01185     }
01186 
01187     gtk_tree_path_free ( path );
01188 
01189     gtk_tree_view_expand_all ( GTK_TREE_VIEW ( priv -> tree_view ) );
01190     gtk_tree_selection_unselect_all ( priv -> selection );
01191 
01192     complete_string = gtk_combofix_get_first_string_from_sort_model ( priv -> model_sort, string );
01193 
01194     return complete_string;
01195 }

01854 
01855     switch (direction)
01856     {
01857         case COMBOFIX_DOWN:

01859             {

01861                      &&

01867                 result = gtk_tree_model_get_iter ( model, iter, path );
01868 
01869                 /* if result is FALSE, perhaps we are on the end of the children list... */
01870                 if ( !result

01872                     gtk_tree_path_get_depth ( path ) > 1 )
01873                 {

01875                     gtk_tree_path_next ( path );

01894                 result = gtk_tree_path_prev ( path );
01895

01899                      * else, stay there */
01900                     result = gtk_tree_model_get_iter ( model, iter, path );

01902                     if ( result
01903                             &&
01904                             gtk_tree_model_iter_has_child ( model, iter )
01905                             &&
01906                             gtk_tree_view_row_expanded ( GTK_TREE_VIEW ( priv -> tree_view ), path ) )
01907                     {
01908                         /* there is some children, go to the last one */
01909                         gint i;
01910 
01911                         gtk_tree_path_down ( path );
01912 
01913                         for ( i = 0 ; i < gtk_tree_model_iter_n_children ( model, iter ) - 1 ; i++ )
01914                             gtk_tree_path_next ( path );
01915 
01916                         result = gtk_tree_model_get_iter ( model, iter, path );
01917                     }
01918                 }
01919                 else
01920                 {
01921                     /* there is no prev path, if we are not on the toplevel, go to the

01923

01925                     {
01926                 gtk_tree_path_up ( path );

01928                     }
01929                 }
