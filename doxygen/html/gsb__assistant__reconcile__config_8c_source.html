00142 GtkResponseType gsb_assistant_reconcile_config_run ( void )
00143 {
00144     GtkResponseType return_value;
00145     GSList *transactions_list = NULL;
00146     GSList *tmp_list;
00147     gint transaction_number;
00148     GtkWidget *assistant;
00149         gchar* tmpstr;
00150 
00151     tmp_list = gsb_data_transaction_get_transactions_list ();
00152     while (tmp_list)
00153     {
00154         transaction_number = gsb_data_transaction_get_transaction_number (tmp_list -> data);
00155 
00156         if ( gsb_data_transaction_get_marked_transaction (transaction_number) == OPERATION_RAPPROCHEE
00157              &&
00158              !gsb_data_transaction_get_reconcile_number (transaction_number))
00159             transactions_list = g_slist_append ( transactions_list,
00160                                                  GINT_TO_POINTER (transaction_number));
00161         tmp_list = tmp_list -> next;
00162     }
00163 
00164     if (!transactions_list)
00165     {
00166         dialogue (_("No marked transactions without reconciliation found !"));
00167         return GTK_RESPONSE_CANCEL;
00168     }
00169 
00170     /* get the number of transactions to associate, we will decrease it for each association */
00171     transactions_to_link = g_slist_length (transactions_list);
00172 
00173     /* come here if we have some orphan transactions
00174      * this can happen by 2 ways :
00175      * for old users of grisbi, before i don't remember what version, there were no reconcile number,
00176      *          the reconciled transactions were juste marked R
00177      * before the 0.6.0, ctrl R didn't permit to choose a reconcile
00178      *
00179      * for the 2nd item, no problem, we show a list of reconciles and the user can choose what reconcile he wants
00180      * for the first item the problem is there is no reconcile number to go with that transactions...
00181      * so we will use the assistant to
00182      *  -permit to create a reconcile directly (not possible normaly in the configuration
00183      *  -permit to choose a reconcile number for each transactions without reconcile
00184      *  - do an automatic find for reconcile, usefull in the first item, when very much
00185      *          transactions without reconcile, but we need to make the old reconciles before,
00186      *          and set the good date for all the reconciles (because grisbi set them automatically
00187      *          at the first update to grisbi 0.6.0 )*/
00188 
00189     /* first, create the assistant */
00190     tmpstr = g_strdup_printf (_("Grisbi found %d marked transactions not associated with a reconciliation number, "
00191                                                        "this can happen for old users of grisbi or from a misuse of the Ctrl-R shortcut.\n\n"
00192                                                        "This assistant will help you make the link between such transactions and a reconciliation.\n\n"
00193                                                        "Before continuing, you should first check if all the dates of the existing reconciliations are good "
00194                                                        "because grisbi will try to guess them not very precisely "
00195                                                        "(you will be able to create new reconciliations in the next step). "
00196                                                        "Previous reconciliations will be available too."),
00197                                                        transactions_to_link );
00198     assistant = gsb_assistant_new ( _("Associate orphan transactions to a reconciliation"),
00199                                     tmpstr,
00200                                     "reconat.png",
00201                                     NULL );
00202     g_free ( tmpstr );
00203 
00204     gsb_assistant_add_page ( assistant,
00205                              gsb_assistant_reconcile_config_page_menu (assistant),
00206                              RECONCILE_ASSISTANT_MENU,
00207                              RECONCILE_ASSISTANT_INTRO,
00208                              RECONCILE_ASSISTANT_NEW_RECONCILE,
00209                              NULL );
00210     gsb_assistant_add_page ( assistant,
00211                              gsb_assistant_reconcile_config_page_new_reconcile (),
00212                              RECONCILE_ASSISTANT_NEW_RECONCILE,
00213                              RECONCILE_ASSISTANT_MENU,
00214                              RECONCILE_ASSISTANT_MENU,
00215                              NULL );
00216     gsb_assistant_add_page ( assistant,
00217                              gsb_assistant_reconcile_config_page_automatically_associate (assistant),
00218                              RECONCILE_ASSISTANT_AUTOMATICALLY_ASSOCIATE,
00219                              RECONCILE_ASSISTANT_MENU,
00220                              RECONCILE_ASSISTANT_MENU,
00221                              G_CALLBACK (gsb_assistant_reconcile_config_update_auto_asso));
00222     gsb_assistant_add_page ( assistant,
00223                              gsb_assistant_reconcile_config_page_manually_associate (assistant),
00224                              RECONCILE_ASSISTANT_MANUALLY_ASSOCIATE,
00225                              RECONCILE_ASSISTANT_MENU,
00226                              RECONCILE_ASSISTANT_MENU,
00227                              G_CALLBACK (gsb_assistant_reconcile_config_update_manu_asso));
00228     gsb_assistant_add_page ( assistant,
00229                              gsb_assistant_reconcile_config_page_success (),
00230                              RECONCILE_ASSISTANT_SUCCESS,
00231                              RECONCILE_ASSISTANT_MENU,
00232                              RECONCILE_ASSISTANT_MENU,
00233                              NULL );
00234     return_value = gsb_assistant_run (assistant);
00235     gtk_widget_destroy (assistant);
00236     return return_value;
00237 }
