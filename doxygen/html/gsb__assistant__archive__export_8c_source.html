00350 static gboolean gsb_assistant_archive_switch ( GtkWidget *assistant,
00351                                                gint new_page )
00352 {
00353     gint archive_number;
00354     GtkTreeModel *model;
00355     const gchar *export_name;
00356     gboolean success = FALSE;
00357     gchar* tmpstr;
00358 
00359     model = gtk_tree_view_get_model (GTK_TREE_VIEW (archive_export_treeview));
00360     archive_number = gsb_assistant_archive_export_get_selected_archive (model);
00361 
00362     switch (new_page)
00363     {
00364         case ARCHIVE_EXPORT_ASSISTANT_INTRO:
00365             break;
00366 
00367         case ARCHIVE_EXPORT_ASSISTANT_CHOOSE:
00368             if (archive_number)
00369                 gsb_assistant_sensitive_button_next (assistant, TRUE);
00370             else
00371                 gsb_assistant_sensitive_button_next (assistant, FALSE);
00372             break;
00373 
00374         case ARCHIVE_EXPORT_ASSISTANT_NAME:
00375             /* if we come here, an archive must have been selected,
00376              * so needn't to check */
00377             tmpstr = g_markup_printf_escaped ( 
00378                         _("<span size=\"x-large\">Exporting the archive : %s</span>"),
00379                         gsb_data_archive_get_name (archive_number));
00380             gtk_label_set_markup ( GTK_LABEL ( archive_export_label ), tmpstr);
00381             g_free ( tmpstr );
00382         /* on remplace les slash des dates par des points
00383          * avant de fixer le nom du fichier par défaut */
00384         tmpstr = ( gchar * )gsb_data_archive_get_name ( archive_number );
00385         tmpstr = my_strdelimit (tmpstr, "/", "." );
00386             gtk_file_chooser_set_current_name ( GTK_FILE_CHOOSER (archive_export_filechooser), tmpstr );
00387             gtk_file_chooser_set_current_folder ( GTK_FILE_CHOOSER (archive_export_filechooser), gsb_file_get_last_path () );
00388             g_free ( tmpstr );
00389             /* need to set the next button to the next function,
00390              * because if the export failed and the user did previous button,
00391              * the next button stay in the close state */
00392             gsb_assistant_change_button_next ( assistant, GTK_STOCK_GO_FORWARD,
00393                                                GTK_RESPONSE_YES );
00394 
00395             break;
00396 
00397         case ARCHIVE_EXPORT_ASSISTANT_SUCCESS:
00398             export_name = gtk_file_chooser_get_filename (GTK_FILE_CHOOSER (archive_export_filechooser));
00399             switch (gtk_combo_box_get_active (GTK_COMBO_BOX(archive_export_combobox)))
00400             {
00401                 case 0:
00402                     /* GSB format */
00403                     /* the gsb_file_save_save_file function write directly, so we need to check before
00404                      * if the file exists */
00405             /* on vérifie juste que l'extension existe */
00406             if ( ! g_strrstr ( export_name, ".gsb" ) )
00407             {
00408                export_name = g_strconcat ( export_name, ".gsb", NULL );
00409             }
00410                     success = gsb_file_util_test_overwrite (export_name)
00411                         &&
00412                         gsb_file_save_save_file ( export_name, conf.compress_backup, archive_number);
00413                     break;
00414 
00415                 case 1:
00416                     /* QIF format */
00417             if ( ! g_strrstr ( export_name, ".qif" ) )
00418             {
00419                export_name = g_strconcat ( export_name, ".qif", NULL );
00420             }
00421                     success = gsb_qif_export_archive ( export_name, archive_number);
00422                     break;
00423 
00424                 case 2:
00425                     /* CSV format */
00426             if ( ! g_strrstr ( export_name, ".csv" ) )
00427             {
00428                export_name = g_strconcat ( export_name, ".csv", NULL );
00429             }
00430                     success = gsb_csv_export_archive ( export_name, archive_number);
00431                     break;
00432             }
00433             /* now success = TRUE or FALSE, show the good final page of assistant */
00434             if (success)
00435                 gtk_label_set_text ( GTK_LABEL (archive_export_label_success),
00436                                      _("Success !\nThe export of the archive finished successfully.\n\nYou can now close the assistant."));
00437             else
00438             {
00439                 gtk_label_set_text ( GTK_LABEL (archive_export_label_success),
00440                                      _("Error !\nAn error occured while saving the archive.\n\nPlease press the Previous button to correct the problem,\nor the close button to cancel the action."));
00441                 gsb_assistant_sensitive_button_prev ( assistant, TRUE );
00442             }
00443     }
00444     return FALSE;
00445 }

00457 static gboolean gsb_assistant_archive_export_toggled ( GtkCellRendererToggle *renderer,
00458                                                        gchar *path_string,
00459                                                        GtkWidget *assistant )
00460 {
00461     GtkTreeIter iter;
00462     gboolean value;
00463     GtkTreePath *path;
00464     GtkTreePath *path_clicked;
00465     GtkTreeModel *model;
00466     gboolean selected = FALSE;
00467 
00468     model = gtk_tree_view_get_model (GTK_TREE_VIEW (archive_export_treeview));
00469     path_clicked = gtk_tree_path_new_from_string (path_string);
00470     path = gtk_tree_path_new_first ();
00471 
00472     while (gtk_tree_model_get_iter ( GTK_TREE_MODEL (model),
00473                                      &iter,
00474                                      path ))
00475     {
00476         /* if we are on the clicked path, we invert the value,
00477          * else, we set the value to 0 */
00478         if (gtk_tree_path_compare (path, path_clicked))
00479             /* the 2 path are different */
00480             value = 0;
00481         else
00482         {
00483             /* the paths are equal */
00484             gtk_tree_model_get ( GTK_TREE_MODEL (model),
00485                                  &iter,
00486                                  ARCHIVES_EXPORT_SELECT_COLUMN, &value,
00487                                  -1 );
00488             value = !value;
00489 
00490             /* we will unsensitive the next button only if something chosen */
00491             if (value)
00492                 selected = TRUE;
00493         }
00494         gtk_list_store_set ( GTK_LIST_STORE (model),
00495                              &iter,
00496                              ARCHIVES_EXPORT_SELECT_COLUMN, value,
00497                              -1 );
00498         gtk_tree_path_next (path);
00499     }
00500     gtk_tree_path_free (path);
00501     gtk_tree_path_free (path_clicked);
00502 
00503     gsb_assistant_sensitive_button_next (assistant, selected);
00504 
00505     return FALSE;
00506 }
