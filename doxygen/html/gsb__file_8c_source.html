00544 gboolean gsb_file_save_file ( gint origine )
00545 {
00546     gint etat_force, result;
00547     gchar *nouveau_nom_enregistrement;
00548 
00549     devel_debug_int (origine);
00550 
00551     etat_force = 0;
00552 
00553     if ( ( !etat.modification_fichier && origine != -2 ) ||
00554          !gsb_data_account_get_accounts_amount () )
00555     {
00556         notice_debug ( "nothing done in gsb_file_save_file" );
00557         return ( TRUE );
00558     }
00559 
00560     /* si le fichier de comptes n'a pas de nom ou si on enregistre sous un nouveau nom */
00561     /*     c'est ici */
00562 
00563     if ( !nom_fichier_comptes || origine == -2 )
00564         nouveau_nom_enregistrement = gsb_file_dialog_ask_name ();
00565     else
00566         nouveau_nom_enregistrement = nom_fichier_comptes;
00567 
00568     if ( !nouveau_nom_enregistrement )
00569         return FALSE;
00570 
00571     /*     on vérifie que le fichier n'est pas locké */
00572     if ( etat.fichier_deja_ouvert
00573          &&
00574          !conf.force_enregistrement
00575          &&
00576          origine != -2 )
00577     {
00578         gchar* tmpstr1 = g_strdup_printf( _("Grisbi was unable to save this file because it is locked.  Please save it with another name or activate the \"%s\" option in preferences."),

00581                                                nom_fichier_comptes );
00582         dialogue_error_hint ( tmpstr1,
00583                               tmpstr2 );
00584         g_free ( tmpstr1 );
00585         g_free ( tmpstr2 );
00586         return ( FALSE );

00589         /* Si le fichier est un d'une version précédente de grisbi on demande si on l'efface */
00590     if ( copy_old_filename && strlen ( copy_old_filename ) > 0 )
00591     {
00592         gsb_file_save_remove_old_file ( copy_old_filename );
00593         g_free ( copy_old_filename );
00594         copy_old_filename = NULL;
00595     }
00596 
00597     /* make backup before saving if asked */
00598     if (conf.make_backup)
00599         gsb_file_save_backup();
00600 
00601     /*   on a maintenant un nom de fichier */

00603     gsb_status_message ( _("Saving file") );
00604 
00605     result = gsb_file_save_save_file ( nouveau_nom_enregistrement,
00606                                        conf.compress_file,
00607                                        FALSE );
00608 
00609     if ( result )
00610     {
00611         /* saving was right, so unlock the last name */
00612         gsb_file_util_modify_lock ( FALSE );
00613

00615 
00616         /* and lock the new name */
00617         gsb_file_util_modify_lock ( TRUE );
00618 
00619         /* update variables */
00620         etat.fichier_deja_ouvert = 0;
00621     modification_fichier ( FALSE );
00622         gsb_file_update_window_title ();
00623         gsb_file_append_name_to_opened_list ( nom_fichier_comptes );
00624     }

00639 gboolean gsb_file_save_backup ( void )
00640 {
00641     gboolean retour;
00642     gchar *filename;
00643     struct tm *day_time;
00644     time_t temps;
00645     gchar *name;
00646 
00647     if (!gsb_file_get_backup_path ()
00648         ||
00649         !etat.modification_fichier )
00650         return FALSE;
00651 
00652     gsb_status_message ( _("Saving backup") );
00653 
00654     name = g_path_get_basename (nom_fichier_comptes);
00655     if ( g_str_has_suffix ( name, ".gsb" ) )
00656     {
00657         gchar **tab_str;
00658 
00659         tab_str = g_strsplit ( name, ".gsb", 0 );
00660         if ( tab_str[0] && strlen ( tab_str[0] ) )
00661         {
00662             g_free ( name );
00663             name = g_strdup ( tab_str[0] );
00664             g_strfreev ( tab_str );
00665         }
00666     }
00667     /* create a filename for the backup :
00668      * filename_yyyymmddTmmhhss.gsb */
00669     if ( conf.make_bakup_single_file )
00670     {
00671         filename =  g_strdup_printf ( "%s%s%s_backup.gsb",
00672                             gsb_file_get_backup_path ( ),
00673                             G_DIR_SEPARATOR_S,
00674                             name );
00675     }
00676     else
00677     {
00678         time ( &temps );
00679         day_time = localtime (&temps);
00680         filename =  g_strdup_printf ( "%s%s%s_%d%02d%02dT%02d%02d%02d.gsb",
00681                             gsb_file_get_backup_path (),
00682                             G_DIR_SEPARATOR_S,
00683                             name,
00684                             day_time -> tm_year + 1900,
00685                             day_time -> tm_mon + 1,
00686                             day_time -> tm_mday,
00687                             day_time -> tm_hour,
00688                             day_time -> tm_min,
00689                             day_time -> tm_sec );
00690     }
00691 
00692     retour = gsb_file_save_save_file ( filename, conf.compress_backup, FALSE );
00693 
00694     g_free (filename);
00695     g_free (name);
00696 
00697     gsb_status_message ( _("Done") );
00698 
00699     return ( retour );
00700 }
