00049 GSList * csv_parse_line ( gchar ** contents, gchar * separator )
00050 {
00051     gchar * tmp = (* contents), * begin = tmp;
00052     gint is_unquoted = FALSE, len = strlen ( separator );
00053     GSList * list = NULL;
00054 
00055     if ( *tmp == '\n' )
00056     {
00057         *contents = tmp+1;
00058         return GINT_TO_POINTER(-1);
00059     }
00060     
00061     if ( *tmp == '!' || *tmp == '#' || *tmp == ';' )
00062     {
00063         *contents = strchr ( tmp, '\n' ) + 1;
00064         return GINT_TO_POINTER(-1);
00065     }
00066 
00067     while ( *tmp )
00068     {
00069         switch ( *tmp )
00070         {
00071             case '\n':
00072                 list = g_slist_append ( list, sanitize_field ( begin, tmp ) );
00073                 *contents = tmp+1;
00074                 return list;
00075 
00076             case '"':
00077                 if ( ! is_unquoted )
00078                 {
00079                     tmp++;
00080                     while ( *tmp )
00081                     {
00082                         /* This is lame escaping but we need to
00083                          * support it. */
00084                         if ( *tmp == '\\' && *(tmp+1) == '"' )
00085                         {
00086                             tmp += 2;
00087                         }
00088 
00089                         /* End of quoted string. */
00090                         if ( *tmp == '"' && *(tmp+1) != '"' )
00091                         {
00092                             break;
00093                         }
00094                         
00095                         tmp++;
00096                     }
00097                 }
00098 
00099             default:
00100                 is_unquoted = TRUE;
00101                 if ( !strncmp ( tmp, separator, len ) )
00102                 {
00103                     list = g_slist_append ( list, sanitize_field ( begin, tmp ) );
00104                     begin = tmp + len;
00105                     is_unquoted = FALSE;
00106                 }
00107                 break;
00108         }
00109 
00110         tmp++;
00111     }
00112 
00113     return NULL;
00114 }

00122 gchar * sanitize_field ( gchar * begin, gchar * end  )
00123 {
00124     gchar * field, * iter;
00125 
00126     g_return_val_if_fail ( begin <= end, NULL );
00127 
00128     if ( end <= begin )
00129         return "";
00130 
00131     iter = field = g_malloc0 ( end - begin + 1 );
00132 
00133     /* Strip out intial white spaces. */
00134     while ( *begin == ' ' )
00135         begin++;
00136 
00137     if ( *begin == '"' )
00138     {
00139         begin++;
00140         while ( *end != '"' && end >= begin )
00141             end--;
00142     }
00143 
00144     while ( begin < end )
00145     {
00146         if ( *begin == '"' && *(begin+1) == '"' )
00147             begin++;
00148 
00149         if ( *begin == '\\' && *(begin+1) == '"' )
00150             begin++;
00151         
00152         *iter++ = *begin++;
00153     }
00154 
00155     /* Strip out remaining white spaces. */
00156     while ( *(iter-1) == ' ' || *(iter-1) == '\n' )
00157         iter--;
00158 
00159     *iter = '\0';
00160 
00161     return field;
00162 }
