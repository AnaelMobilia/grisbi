00244 gint gsb_form_transactions_look_for_last_party ( gint no_party,
00245                         gint no_new_transaction,
00246                         gint account_number )
00247 {
00248     GSList *list_tmp_transactions;
00249     gint last_transaction_with_party_in_account = 0;
00250     gint last_transaction_with_party_not_in_account = 0;
00251 
00252     list_tmp_transactions = gsb_data_transaction_get_complete_transactions_list ();
00253 
00254     while ( list_tmp_transactions )
00255     {
00256         gint transaction_number_tmp;
00257         transaction_number_tmp = gsb_data_transaction_get_transaction_number (list_tmp_transactions -> data);
00258 
00259         if ( gsb_data_transaction_get_party_number (transaction_number_tmp) == no_party
00260              &&
00261              transaction_number_tmp != no_new_transaction
00262              &&
00263              !gsb_data_transaction_get_mother_transaction_number (transaction_number_tmp))
00264         {
00265             /* we are on a transaction with the same party, it's also a split, so we keep it */
00266             if ( gsb_data_transaction_get_account_number (transaction_number_tmp) == account_number)
00267                 last_transaction_with_party_in_account = transaction_number_tmp;
00268             else
00269                 last_transaction_with_party_not_in_account = transaction_number_tmp;
00270         }
00271         list_tmp_transactions = list_tmp_transactions -> next;
00272     }
00273 
00274     if ( last_transaction_with_party_in_account )
00275         return last_transaction_with_party_in_account;
00276 
00277     /* if we don't want to complete with a transaction in another account,
00278      * go away here */
00279     if ( etat.limit_completion_to_current_account )
00280         return 0;
00281 
00282     return last_transaction_with_party_not_in_account;

00426 gint gsb_form_transaction_validate_transfer ( gint transaction_number,
00427                         gint new_transaction,
00428                         gint account_transfer )
00429 {
00430     gint contra_transaction_number;
00431     gint contra_mother_number = 0;
00432     gint contra_marked_transaction = 0;
00433     GtkWidget *contra_payment_button;
00434 
00435     g_return_val_if_fail ( account_transfer >= 0, -1 );
00436 
00437     /* either it's a new transfer or a change of a non-transfer transaction
00438      * either it was already a transfer, in that case, if we change the target account,
00439      * we delete the contra-transaction and it's the same as a new transfer */
00440     if ( !new_transaction )
00441     {
00442         /* it's a modification of a transaction */
00443 
00444         /* as we will do a transfer, the category number is null */
00445         gsb_data_transaction_set_category_number ( transaction_number, 0 );
00446         gsb_data_transaction_set_sub_category_number ( transaction_number, 0 );
00447         contra_transaction_number = gsb_data_transaction_get_contra_transaction_number (
00448                         transaction_number);
00449 
00450         if (contra_transaction_number > 0)
00451         {
00452             /* the transaction is a transfer */
00453 
00454             /* if the contra transaction was a child of split, copying/deleting it
00455              * will remove the information of the mother, so we get it here */
00456             contra_mother_number = gsb_data_transaction_get_mother_transaction_number (
00457                         contra_transaction_number );
00458 
00459         /* Copying/deleting remove the marked information, so we get it here */
00460         contra_marked_transaction = gsb_data_transaction_get_marked_transaction (
00461                         contra_transaction_number );
00462         
00463             /* check if we change the account target */
00464             if ( gsb_data_transaction_get_contra_transaction_account (
00465                         transaction_number) != account_transfer )
00466             {
00467             /* it was a transfer and the user changed the target account so we delete
00468              * the last contra transaction contra_transaction_transfer has just been set */
00469 
00470             gsb_data_transaction_set_contra_transaction_number (
00471                             contra_transaction_number, 0);
00472             gsb_transactions_list_delete_transaction (contra_transaction_number, FALSE);
00473             new_transaction = 1;
00474             }
00475         }
00476         else
00477         {
00478             /* the transaction was not a transfer, so it's the same as a new transaction, to do the contra-transaction */
00479 
00480             new_transaction = 1;
00481         }
00482     }
00483 
00484     /* so, now, it's either a new transfer and new_transaction is TRUE,
00485      * either a transfer without changing the target account and in that case, contra_transaction_number is
00486      * already set */
00487 
00488     if ( new_transaction )
00489         contra_transaction_number = gsb_data_transaction_new_transaction ( account_transfer );
00490 
00491     gsb_data_transaction_copy_transaction ( transaction_number,
00492                                             contra_transaction_number, new_transaction );
00493 
00494     /* contra_mother_number contains the mother number of the contra transaction if it was a modification
00495      * and that contra-transaction was a child of split, and if not it is 0, and it's a good thing because
00496      * now contra-transaction has a mother number if the transaction was a child of split, and we have to
00497      * set that mother number to 0 (for the contra-transaction) */
00498     gsb_data_transaction_set_mother_transaction_number ( contra_transaction_number,
00499                                                          contra_mother_number );
00500 
00501     /* If this is not a new transaction it restores the marked statement */
00502     gsb_data_transaction_set_marked_transaction ( contra_transaction_number,
00503                         contra_marked_transaction );
00504 
00505     /* we have to change the amount by the opposite */
00506     gsb_data_transaction_set_amount (contra_transaction_number,
00507                                      gsb_real_opposite (gsb_data_transaction_get_amount (transaction_number)));
00508 
00509 
00510     /* we have to check the change */
00511     gsb_currency_check_for_change ( contra_transaction_number );
00512 
00513     /* set the contra-method of payment,
00514      * there is no place into the transaction structure for that, so it was not taken with the form,
00515      * we need to get it now from the form */
00516     contra_payment_button = gsb_form_widget_get_widget (TRANSACTION_FORM_CONTRA);
00517     if (contra_payment_button
00518         &&
00519         GTK_WIDGET_VISIBLE (contra_payment_button))
00520         gsb_data_transaction_set_method_of_payment_number ( contra_transaction_number,
00521                                                             gsb_payment_method_get_selected_number (contra_payment_button));
00522 
00523     /* set the link between the transactions */
00524     gsb_data_transaction_set_contra_transaction_number ( transaction_number,
00525                                                          contra_transaction_number);
00526     gsb_data_transaction_set_contra_transaction_number ( contra_transaction_number,
00527                                                          transaction_number);
00528 
00529     /* show the contra_transaction */
00530     if ( new_transaction )
00531         gsb_transactions_list_append_new_transaction (contra_transaction_number, TRUE);
00532     else
00533         gsb_transactions_list_update_transaction (contra_transaction_number);
00534 
00535     return contra_transaction_number;
00536 }
