00299     gint return_value;
00300     gchar *file_content;
00301     gulong length;
00302 
00303     devel_debug ( filename );
00304 
00305     /* general check */
00306     
00307     if ( !g_file_test (filename, G_FILE_TEST_EXISTS) )
00308     {
00309         gchar* tmpstr1 = g_strdup_printf ( _("Cannot open file '%s': %s"),

00321     if ( !g_file_test ( filename, G_FILE_TEST_IS_REGULAR ))
00322     {
00323         gchar* tmpstr1 = g_strdup_printf ( 
00324                         _("%s doesn't seem to be a regular file,\nplease check it and try again."),

00330         gsb_file_remove_name_from_opened_list (filename);
00331         return ( FALSE );
00332     }
00333 
00334      /* fill the buffer stat to check the permission */
00335     return_value = g_stat (filename, &buffer_stat);
00336     /* check the access to the file and propose to change it */
00337 #ifndef _WIN32

00340 #endif /* _WIN32 */
00341

00344     {

00346         GMarkupParseContext *context;
00347         gsb_plugin *plugin;
00348 
00349         /* first, we check if the file is crypted, if it is, we decrypt it */
00350         if ( !strncmp ( file_content, "Grisbi encrypted file ", 22 ) )
00351         {
00352             plugin = gsb_plugin_find ( "openssl" );
00353             if ( plugin )

00360             if ( ! length )

00370                         _("Encryption plugin not found." ) );
00371                 return FALSE;

00376          * because the xml structure changes after 0.6 */

00379         {
00380             /* fill the GMarkupParser for a new xml structure */
