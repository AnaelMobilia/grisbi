00073 void gsb_data_print_config_init (void)
00074 {
00075     /* init transactions list config */
00076     draw_lines = TRUE;
00077     draw_column = TRUE;
00078     draw_background = FALSE;
00079     draw_archives = FALSE;
00080     draw_columns_name = TRUE;
00081     draw_title = FALSE;
00082 
00083     draw_interval_dates = FALSE;
00084     draw_dates_are_value_dates = FALSE;
00085 
00086     if (font_transactions)
00087         pango_font_description_free (font_transactions);
00088     if ( conf.utilise_fonte_listes && conf.font_string )
00089         font_transactions = pango_font_description_from_string ( conf.font_string );
00090     else
00091         font_transactions = pango_font_description_from_string  ("sans 6");
00092 
00093     if (font_title)
00094         pango_font_description_free (font_title);
00095     font_title = pango_font_description_from_string  ("sans 12");
00096 
00097     /* init report config */
00098     if (report_font_transactions)
00099         pango_font_description_free (report_font_transactions);
00100     report_font_transactions = pango_font_description_from_string  ("sans 6");;
00101 
00102     if (report_font_title)
00103         pango_font_description_free (report_font_title);
00104     report_font_title = pango_font_description_from_string  ("sans 12");
00105 }
