00223                         gint bank_number,

00225 {
00226     GtkTreeIter iter;

00231     if (!gtk_tree_model_get_iter_first ( GTK_TREE_MODEL (bank_list_model),

00253 
00254         if (bank_number == current_bank)
00255         {
00256             /* bank found */
00257             gtk_combo_box_set_active_iter ( GTK_COMBO_BOX (combobox),

00259             break;

00261     }
00262     while (gtk_tree_model_iter_next (GTK_TREE_MODEL (bank_list_model), &iter));
00263 
00264     g_object_set_data ( G_OBJECT (combobox),

00269         g_signal_handler_unblock ( G_OBJECT(combobox),
00270                                    (gulong) g_object_get_data (G_OBJECT (combobox),

00273         g_signal_handler_unblock ( G_OBJECT(combobox),
00274                                    (gulong) g_object_get_data (G_OBJECT (combobox),
00275                                                                "changed-hook"));

01156 static gboolean gsb_bank_delete ( GtkWidget *button,
01157                         gpointer null )
01158 {
01159     gint bank_number;
01160     gboolean result;
01161     gboolean bank_is_used=FALSE;
01162     GSList *list_tmp;
01163     GtkTreeSelection *selection;
01164     GtkTreeModel *model;
01165     GtkTreeIter iter;
01166 
01167     selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (bank_list_tree_view));
01168     if ( !gtk_tree_selection_get_selected ( GTK_TREE_SELECTION (selection),
01169                                             &model, &iter ))
01170         return FALSE;
01171 
01172     /* get the bank number */
01173     gtk_tree_model_get ( GTK_TREE_MODEL (model),
01174                          &iter,
01175                          BANK_NUMBER, &bank_number,
01176                          -1 );
01177 
01178     /* check if the bank is used in grisbi */
01179     list_tmp = gsb_data_account_get_list_accounts ();
01180 
01181     while ( list_tmp )
01182     {
01183         gint i;
01184 
01185         i = gsb_data_account_get_no_account ( list_tmp -> data );
01186 
01187         if ( gsb_data_account_get_bank (i) == bank_number )
01188             bank_is_used = TRUE;
01189 
01190         list_tmp = list_tmp -> next;
01191     }
01192 
01193     if ( bank_is_used )
01194     {
01195         gchar* tmpstr = g_strdup_printf ( _("Bank \"%s\" is used by one or several accounts.\nDo you really want to remove it?"),
01196                                                           gsb_data_bank_get_name (bank_number));
01197         result = question_yes_no_hint ( _("Confirmation of bank removal"),
01198                                         tmpstr,
01199                                         GTK_RESPONSE_NO );
01200        g_free ( tmpstr );
01201     }
01202     else
01203     {
01204         gchar* tmpstr = g_strdup_printf ( _("Are you sure you want to remove bank \"%s\"?\n"),
01205                                                           gsb_data_bank_get_name (bank_number));
01206         result = question_yes_no_hint ( _("Confirmation of bank removal"),
01207                                         tmpstr,
01208                                         GTK_RESPONSE_NO );
01209        g_free ( tmpstr );
01210     }
01211 
01212     if ( result )
01213     {
01214         /* remove the bank */
01215         gtk_list_store_remove ( GTK_LIST_STORE (model),
01216                                 &iter );
01217         gsb_data_bank_remove (bank_number);
01218         if ( etat.modification_fichier == 0 )
01219         modification_fichier ( TRUE );
01220     }
01221 
01222     /* remove the bank from the combobox list */
01223     gsb_bank_update_selected_line_model (NULL);
01224 
01225     if ( etat.modification_fichier == 0 )
01226         modification_fichier ( TRUE );
01227     return FALSE;
01228 }
