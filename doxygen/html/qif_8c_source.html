00598      * and check if all the dates are possible with that order. if one day 2 different order
00599      * can be good with that check, we should implement a second check : the transactions
00600      * are sorted normally, either in ascending, sometimes in descending order.
00601      * so check for the valids order if the transactions are sorted, normally, only one we be correct */
00602 
00603     tmp_list = transactions_list;
00604     while (tmp_list)
00605     {
00606         struct struct_ope_importation *transaction = tmp_list -> data;

00609 
00610         if (!transaction -> date_tmp)
00611             continue;

00616         if (memchr (array[2], '/', strlen (array[2])))
00617         {
00618             gchar *string = g_strdup_printf ( _("The date %s seems contains more than 2 separators.\nThis shouldn't happen. Please contact the Grisbi team to try to add your strange format into Grisbi"),
