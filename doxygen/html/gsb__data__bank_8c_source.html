00116 
00117     if (!bank_number)
00118         return NULL;
00119 
00120     /* before checking all the banks, we check the buffer */

00123          bank_buffer -> bank_number == bank_number )

00126     tmp = bank_list;

00128     while ( tmp )

00132         bank = tmp -> data;
00133 
00134         if ( bank -> bank_number == bank_number )

00269 gboolean gsb_data_bank_remove ( gint bank_number )
00270 {
00271     struct_bank *bank;
00272     GSList *list_tmp;
00273 
00274     bank = gsb_data_bank_get_structure ( bank_number );
00275 
00276     if (!bank)
00277         return FALSE;
00278 
00279     bank_list = g_slist_remove ( bank_list,
00280                                  bank );
00281     _gsb_data_bank_free ( bank );
00282 
00283     /* remove that bank of the accounts */
00284     list_tmp = gsb_data_account_get_list_accounts ();
00285 
00286     while (list_tmp)
00287     {
00288         gint account_number = gsb_data_account_get_no_account (list_tmp -> data);
00289 
00290         if ( gsb_data_account_get_bank (account_number) == bank_number )
00291             gsb_data_account_set_bank ( account_number, 0 );
00292         list_tmp = list_tmp -> next;
00293     }
00294 
00295     return TRUE;
00296 }
