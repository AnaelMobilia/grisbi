00072  * Creates a new GtkEntry with a pointer to a string that will be
00073  * modified according to the entry's value.
00074  *
00075  * \param value A pointer to a string
00076  * \param hook An optional function to execute as a handler if the
00077  *      entry's contents are modified. (hook must be func ( GtkWidget *entry, gpointer data ) )
00078  * \param data An optional pointer to pass to hooks.
00079  */
00080 GtkWidget *gsb_automem_entry_new ( gchar **value,
00081                                    GCallback hook,
00082                                    gpointer data )
00083 {
00084     GtkWidget *entry;
00085 
00086     entry = gtk_entry_new ();
00087 
00088     if (value && *value)
00089         gtk_entry_set_text ( GTK_ENTRY(entry),
00090                              *value );
00091 
00092     g_object_set_data ( G_OBJECT ( entry ),
00093                         "pointer", value);
00094     g_object_set_data ( G_OBJECT (entry), "changed",
00095                         (gpointer) g_signal_connect_after ( G_OBJECT(entry), "changed",
00096                                                             G_CALLBACK (gsb_automem_entry_changed), NULL));
00097     if (hook)
00098         g_object_set_data ( G_OBJECT (entry), "changed-hook",
00099                             (gpointer) g_signal_connect_after ( G_OBJECT(entry), "changed",
00100                                                                 G_CALLBACK (hook), data));
00101     return entry;
00102 }
